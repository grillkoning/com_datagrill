1186  function getForeignKeyValues($values = null, $sql = null, $record = null){

1188  if ( !isset($values) and !isset($sql) and !isset($record) and isset($this->_cache[__FUNCTION__]) ){
1189  return $this->_cache[__FUNCTION__];
1190  }
1191  if ( !isset($values) and !isset($sql) and is_object($record) and isset($this->_cache[__FUNCTION__.$record_id]) ){
1192  return $this->_cache[__FUNCTION__.$record_id];
1193  }
1194  // Strategy:
1195  // ----------
1196  // 1. Label all fields involved in the foreign key so that fields that are equal have the
1197  // same label. Eg: In the query:
1198  // select *
1199  // from Students
1200  // inner join Student_Courses
1201  // on Students.id = Student_Courses.studentid
1202  // inner join Courses
1203  // on Student_Courses.courseid = Courses.id
1204  // where
1205  // Students.id = '$id'
1206  //
1207  // In the above query Students.id and Student_Course.studentid would have the same label, and
1208  // Student_Courses.courseid and Courses.id would have the same label.
1209  // ie: Label(Students.id) = Label(Student_Courses.studentid) ^ Label(Student_Courses.courseid) = Label(Courses.id)
1210  //
1211  // 2. Assign values for each label. All fields with a particular label have the same value.
1212  // In the above query, we would have:
1213  // Value(Label(Students.id)) = '$id'
1214  // **Note from above that Label(Students.id)=Label(Student_Courses.studentid) so their values are also equal.
1215  //
1216  // 3. For labels without a value find out if one of the fields assuming that label is an auto-increment field. If so
1217  // we assign the special value '__Tablename__auto_increment__' where "Tablename" is the name of the table whose
1218  // field is to be auto incremented.
1219  //
1220  // 4. Collect the the values in a structure so that we can lookup the values of any particular field in any particular
1221  // table easily. Return this structure.
1222  $relationship =& $this->_schema;

1226  // for the rest of the computations.

1247

1268  // a special value (in this case), so that we know this is the case.
