1 (function(){
  2 	var $ = jQuery;
  3 	XataJax.ui.application.Tool = Tool;
  4 	XataJax.ui.application.ToolListener = ToolListener;
  5 	
  6 	/**
  7 	 * Error code used to cancel a selection or deselection action
  8 	 * silently.  This code can be used with exceptions thrown
  9 	 * from the beforeSelect or beforeDeselect methods of a 
 10 	 * ToolListener to quietly cancel the selection or deselection.
 11 	 */
 12 	XataJax.errorcodes.CANCEL_ACTION = XataJax.nextErrorCode();
 13 	
 14 	function Tool(o){
 15 		$.extend(this, {
 16 			setName: setName,
 17 			getName: getName,
 18 			setLabel: setLabel,
 19 			getLabel: getLabel,
 20 			setTooltip: setTooltip,
 21 			getTooltip: getTooltip,
 22 			setIcon: setIcon,
 23 			getIcon: getIcon,
 24 			setCssClass: setCssClass,
 25 			getCssClass: getCssClass,
 26 			getElement: getElement,
 27 			select: select,
 28 			deselect: deselect,
 29 			isSelected: isSelected,
 30 			addListener: addListener,
 31 			removeListener: removeListener
 32 			
 33 		});
 34 		var self = this;
 35 	
 36 		var members = {
 37 			name: null,
 38 			label: null,
 39 			tooltip: null,
 40 			icon: null,
 41 			listeners: [],
 42 			cssClass: null,
 43 			el: null
 44 		};
 45 		$.extend(members, o);
 46 		
 47 		
 48 		function setName(name){
 49 			members.name = name;
 50 		}
 51 		
 52 		function getName(){
 53 			return members.name;
 54 		}
 55 		
 56 		function setLabel(label){
 57 			return members.label = label;
 58 		}
 59 		
 60 		function getLabel(){
 61 			if ( members.label != null ) return members.label;
 62 			else return members.name;
 63 		}
 64 		
 65 		function setTooltip(tt){
 66 			members.tooltip = tt;
 67 		}
 68 		
 69 		function getTooltip(){
 70 		
 71 			return members.tooltip;
 72 		}
 73 		
 74 		/**
 75 		 * Sets the icon to use on this tool's button.
 76 		 * @param string icon
 77 		 */
 78 		function setIcon(icon){
 79 			members.icon = icon;
 80 		}
 81 		
 82 		
 83 		/**
 84 		 * Returns the icon to use on this tool's button.
 85 		 *  @return string
 86 		 */
 87 		function getIcon(){
 88 			return members.icon;
 89 		}
 90 		
 91 		
 92 		/**
 93 		 * @return string
 94 		 */
 95 		function getCssClass(){
 96 			return members.cssClass;
 97 		}
 98 		
 99 		
100 		/**
101 		 * Sets the CSS class to be used on the DOMElement button for this tool.
102 		 * @param string css
103 		 */
104 		function setCssClass(css){
105 			members.cssClass = css;
106 		}
107 		
108 		/**
109 		 * Returns the DOM element that serves as this tool's ui button.
110 		 * @returns DOMElement
111 		 */
112 		function getElement(){
113 			if ( members.el == null ){
114 				members.el = document.createElement('a');
115 				$(members.el).addClass('xatajax-application-tool');
116 				if ( members.cssClass != null ) $(members.el).addClass(members.cssClass);
117 				if ( members.icon != null ) $(members.el).css('background-image', 'url('+members.icon+')');
118 				
119 				$(members.el).click(fireElementClicked);
120 			}
121 			return members.el;
122 		}
123 		
124 		/**
125 		 * Listener method that is called whenever the tool's button is clicked.
126 		 */
127 		 
128 		function fireElementClicked(event){
129 			select(event);
130 		}
131 		
132 		
133 		/**
134 		 * Selects the tool.  This method will first call the beforeSelect
135 		 * method of all listeners to give them an opportunity to override
136 		 * the selection.  Any exception thrown from the beforeSelect
137 		 * method with the code XataJax.ui.application.errorcodes.CANCEL_ACTION
138 		 * will cause the selection to not occur.  After the select has successfully
139 		 * completed this will call the onSelect method of the listeners.
140 		 */
141 		function select(event){
142 			if ( $(members.el).hasClass('selected') ) 
143 			try {
144 				$.each(members.listeners, function(){
145 					($.proxy(this.beforeSelect,self))(event);
146 				});
147 			} catch (e){
148 				e = new XataJax.Exception(e);
149 				switch (e.getCode()){
150 					case XataJax.errorcodes.CANCEL_ACTION:
151 						return;
152 					default:
153 						alert(e.getMessage());
154 				}
155 				
156 			}
157 			
158 			$(members.el).addClass('selected');
159 			$.each(members.listeners, function(){
160 				($.proxy(this.onSelect, self))(event);
161 			});
162 		
163 		}
164 		
165 		
166 		/**
167 		 * Deselects the tool.  This method will first call the listeners' 
168 		 * beforeDeselect method to give them an opportunity to override the
169 		 *	event.  Any exception thrown from the beforeDeselect method
170 		 * 	with the code XataJax.ui.application.errorcodes.CANCEL_ACTION will
171 		 * result in this method not completing the deselect.
172 		 *
173 		 * @param eventdata event The jquery event data.  May be undefined.  This is optional.
174 		 * @return void
175 		 */
176 		function deselect(event){
177 			if ( !$(members.el).hasClass('selected') ) return;
178 			try {
179 				$.each(members.listeners, function(){
180 					($.proxy(this.beforeDeselect, self))(event);
181 				});
182 			} catch (e){
183 				e = new XataJax.Exception(e);
184 				switch (e.getCode()){
185 					case XataJax.errorcodes.CANCEL_ACTION:
186 						return;
187 					default:
188 						alert(e.getMessage());
189 				}
190 				
191 			}
192 			
193 			$(members.el).removeClass('selected');
194 			$.each(members.listeners, function(){
195 				($.proxy(this.onDeselect, self)(event));
196 			});
197 		}
198 		
199 		function isSelected(){
200 			return $(members.el).hasClass('selected');
201 		}
202 		
203 		/**
204 		 * Adds a listener to the tool.
205 		 * @param ToolListener l
206 		 */
207 		function addListener(l){
208 			members.listeners.push(l);
209 		}
210 		
211 		
212 		/**
213 		 * Removes a listener.
214 		 * @param ToolListener l
215 		 */
216 		function removeListener(l){
217 			var idx = members.listeners.indexOf(l);
218 			if ( idx != -1 ) members.listeners.splice(idx,1);
219 		}
220 		
221 		
222 		
223 	}
224 	
225 	
226 	
227 	function ToolListener(o){
228 		$.extend(this, o);
229 	}
230 	
231 	ToolListener.prototype = {
232 		beforeSelect: function(){},
233 		beforeDeselect: function(){},
234 		onSelect: function(){},
235 		onDeselect: function(){}
236 	};
237 	
238 	
239 	
240 })();
