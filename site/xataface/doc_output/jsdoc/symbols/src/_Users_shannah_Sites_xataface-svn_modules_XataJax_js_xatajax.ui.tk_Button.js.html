  6 (function(){
  7 	var $ = jQuery;
  8 	var Component = XataJax.ui.tk.Component;
  9 	
 10 	
 11 	XataJax.ui.tk.Button = Button;
 12 	
 13 	/**
 14 	 * A button UI element.  Buttons can be added to toolbars or other 
 15 	 * parts of the interface.
 16 	 *
 17 	 * Much of this implementation is copied from The Filament Group's
 18 	 * Demonstration at <a href="http://www.filamentgroup.com/lab/styling_buttons_and_toolbars_with_the_jquery_ui_css_framework/>Here</a>.
 19 	 *
 20 	 *
 21 	 * @constructor
 22 	 *
 23 	 * @event {void} click Fired when the button is clicked.
 24 	 */
 25 	function Button(o){
 26 	
 27 	
 28 		/**
 29 		 * @type {String}
 30 		 */
 31 		var label = 'Button';
 32 		
 33 		
 34 		/**
 35 		 * The priority of this button (either primary or secondary)
 36 		 * @type {String}
 37 		 */
 38 		var priority = 'primary';
 39 		
 40 		
 41 		/**
 42 		 * Sets whether the button is disabled.
 43 		 * @type {boolean}
 44 		 */
 45 		var disabled = false;
 46 		
 47 		/**
 48 		 * The icon to use for this button.  This is the name of an icon.
 49 		 * Generally you can choose from the icons in the jQueryUI theme.
 50 		 * This string only holds the name of the icon (not the ui-icon- prefix).
 51 		 *
 52 		 * @type {String}
 53 		 */
 54 		var icon = null;
 55 		
 56 		
 57 		/**
 58 		 * The icon style.  This is one of 'solo', 'left', 'right'.  This 
 59 		 * determines where the icon is displayed in the button.
 60 		 *
 61 		 * @type {String}
 62 		 */
 63 		var iconStyle = 'left';
 64 		
 65 		
 66 		
 67 		/**
 68 		 * A setting to set whether this button is a toggle button.  By 
 69 		 * default this is false.
 70 		 *
 71 		 * @type {boolean}
 72 		 */
 73 		var toggleable = false;
 74 	
 75 		XataJax.extend(this, new Component(o));
 76 		XataJax.publicAPI(this, {
 77 			setLabel: setLabel,
 78 			getLabel: getLabel,
 79 			setIcon: setIcon,
 80 			getIcon: getIcon,
 81 			setIconStyle: setIconStyle,
 82 			setIconStyleLeft: setIconStyleLeft,
 83 			setIconStyleRight: setIconStyleRight,
 84 			setIconStyleSolo: setIconStyleSolo,
 85 			getIconStyle: getIconStyle,
 86 			setPriority: setPriority,
 87 			getPriority: getPriority,
 88 			setDisabled: setDisabled,
 89 			isDisabled: isDisabled,
 90 			createElement: createElement,
 91 			decorateElement: decorateElement
 92 		});
 93 		
 94 		function setToggleable(t){
 95 			if ( t != toggleable ){
 96 				var old = toggleable;
 97 				toggleable = t;
 98 				
 99 				if ( t ){
100 					$(this.getElement()).addClass('xataface-button-toggleable');
101 				} else {
102 					$(this.getElement()).removeClass('xataface-button-toggleable');
103 				}
104 				
105 				this.firePropertyChange('toggleable', old, t);
106 			}
107 		
108 		
109 		}
110 		
111 		/**
112 		 * Sets the icon for this button.
113 		 *
114 		 * See <a href="http://www.themeroller.com/">ThemeRoller</a> for a list of available
115 		 * 	buttons.
116 		 *
117 		 * @param {String} i The jQueryUI icon name.
118 		 * @returns {Button} Self for chaining.
119 		 */
120 		function setIcon(i){
121 			if ( i != icon ){
122 				var old = icon;
123 				icon = i;
124 				var iconEl = $('span.ui-icon', this.getElement());
125 				if ( iconEl.size() == 0 ){
126 					iconEl = document.createElement('span');
127 					iconEl = $(iconEl);
128 					iconEl.addClass('ui-icon');
129 					$(this.getElement()).prepend(iconEl);
130 				}
131 				if ( old ){
132 					iconEl.removeClass('ui-icon-'+old);
133 				}
134 				if ( i ){
135 					iconEl.addClass('ui-icon-'+i);
136 				} else {
137 					iconEl.remove();
138 				}
139 						
140 				this.firePropertyChange('icon', old, i);
141 			}
142 			return this;
143 		}
144 		
145 		
146 		/**
147 		 * Gets the icon for this button.
148 		 *
149 		 * See <a href="http://www.themeroller.com/">Themeroller</a> for a list of available
150 		 * icons.
151 		 *
152 		 * @returns {String} The jQueryUI icon name.
153 		 */
154 		function getIcon(){
155 			return icon;
156 		}
157 		
158 		/**
159 		 * Sets button's icon style.  Either left, right, or solo.
160 		 *
161 		 * @param {String} style Either 'left', 'right', or 'solo'.
162 		 * @returns {Button} Self for chaining.
163 		 */
164 		function setIconStyle(style){
165 			if ( style != iconStyle ){
166 				var old = iconStyle;
167 				iconStyle = style;
168 				
169 				$(this.getElement()).removeClass('xatajax-button-icon-'+old);
170 				$(this.getElement()).addClass('xatajax-button-icon-'+style);
171 				
172 				this.firePropertyChange('iconStyle', old, style);
173 			}
174 			return this;
175 		}
176 		
177 		/**
178 		 * Returns the icon style (left, right, or solo).
179 		 * @returns {String}
180 		 */
181 		function getIconStyle(){
182 			return iconStyle;
183 		}
184 		
185 		
186 		/**
187 		 * Sets the icon style to solo so that the label is hidden and only the icon
188 		 * is displayed.
189 		 * @returns {Button} Self for chaining
190 		 */
191 		function setIconStyleSolo(){
192 			return this.setIconStyle('solo');
193 		}
194 		
195 		/**
196 		 * Sets the icon style to 'left' so that the icon appears to the left of the 
197 		 * button label.
198 		 *
199 		 * @returns {Button} Self for chaining.
200 		 *
201 		 */
202 		function setIconStyleLeft(){
203 			return this.setIconStyle('left');
204 		}
205 		
206 		
207 		/**
208 		 * Sets the icon style to 'right' so that the icon is to the right of the 
209 		 * label.
210 		 *
211 		 * @returns {Button} Self for chaining.
212 		 */
213 		function setIconStyleRight(){
214 			return this.setIconStyle('right');
215 		}
216 		
217 		
218 		/**
219 		 * Sets whether this button is disabled.
220 		 * @param {boolean} d The disabled setting.  True for disabled.
221 		 * @returns {Button} Self for chaining.
222 		 */
223 		function setDisabled(d){
224 			if ( d != disabled ){
225 				var old = disabled;
226 				disabled = d;
227 				if ( d ){
228 					$(this.getElement()).addClass('ui-state-disabled');
229 				} else {
230 					$(this.getElement()).removeClass('ui-state-disabled');
231 				}
232 				this.firePropertyChange('disabled', old, d);
233 			}
234 			return this;
235 		}
236 		
237 		
238 		/**
239 		 * Checks whether button is disabled.
240 		 * @returns {boolean}
241 		 */
242 		function isDisabled(){
243 			return disabled;
244 		}
245 		
246 		/**
247 		 * Sets the label of this button.
248 		 * @param {String} l The string label.
249 		 * @returns {Button} Self for chaining.
250 		 */
251 		function setLabel(l){
252 			if ( label != l ){
253 				var old = label;
254 				label = l;
255 				var iconEl = $('span.ui-icon', this.getElement());
256 				$(this.getElement()).text(l);
257 				if ( iconEl.size() > 0 ){
258 					$(this.getElement()).prepend(iconEl);
259 				}
260 				this.firePropertyChange('label', old, l);
261 			}
262 			return this;
263 		}
264 		
265 		
266 		/**
267 		 * Gets the label of this button.
268 		 * @returns {String}
269 		 */
270 		function getLabel(){
271 			return label;	
272 		}
273 		
274 		/**
275 		 * Sets the priority of the button (either "primary" or "secondary").
276 		 * @param {String} p
277 		 * @returns {Button} Self for chaining.
278 		 */
279 		function setPriority(p){
280 			if ( ['primary','secondary'].indexOf(p) == -1 ){
281 				throw new Exception({
282 					message: 'Illegal parameters for setPriority.  Must be one of "primary" or "secondary"'
283 				});
284 			}
285 			if ( p != priority ){
286 				var old = priority;
287 				priority = p;
288 				$(this.getElement())
289 					.removeClass('ui-priority-'+old)
290 					.addClass('ui-priority-'+p);
291 				this.firePropertyChange('priority', old, p);
292 			}
293 			return this;
294 		}
295 		
296 		/**
297 		 * Gets the priority of the button.  This is either 'primary' or 'secondary'.
298 		 * This can help indicate to the user which button's have a higher priority.  For
299 		 * example in a dialog you may want to emphasize the "save" button over the "cancel"
300 		 * button.  In this case you would set the priority of the "save" button to "primary"
301 		 * and the priority of the "cancel" button to "secondary".
302 		 *
303 		 * @returns {String}
304 		 */
305 		function getPriority(){
306 			return priority;
307 		}
308 		
309 		
310 		
311 		/**
312 		 * Creates the button element.
313 		 * @override
314 		 * @returns {HTMLElement}
315 		 */
316 		function createElement(){
317 			return document.createElement('button');
318 		}
319 		
320 		/**
321 		 * This function is a private function that is meant
322 		 * to be attached to the button's HTMLElement as a
323 		 * listener to the hover event.
324 		 *
325 		 * The "this" context will be the HTMLElement and not the 
326 		 * component.
327 		 *
328 		 */
329 		function setButtonElementHover(){
330 			if ( !$(this).hasClass('ui-state-disabled') ){
331 				$(this).addClass("ui-state-hover");
332 			}
333 		}
334 		
335 		
336 		/**
337 		 * This function is a private function that is meant
338 		 * to be attached to the button's HTMLElement as a
339 		 * listener to the hover event.
340 		 *
341 		 * The "this" context will be the HTMLElement and not the 
342 		 * component.
343 		 *
344 		 */
345 		function clearButtonElementHover(){
346 			if ( !$(this).hasClass('ui-state-disabled') ){
347 				$(this).removeClass("ui-state-hover");
348 			}
349 		}
350 		
351 		
352 		function buttonMouseDown(){
353 			if ( !$(this).hasClass('ui-state-disabled') ){
354 			
355 				if( $(this).is('.ui-state-active.xatajax-button-toggleable') ){ 
356 					$(this).removeClass("ui-state-active"); 
357 				}
358 				else { 
359 					$(this).addClass("ui-state-active"); 
360 				}	
361 			}
362 			
363 		}
364 		
365 		function buttonMouseUp(){
366 			if ( !$(this).hasClass('ui-state-disabled') ){
367 				if(! $(this).is('.xatajax-button-toggleable') ){
368 					$(this).removeClass("ui-state-active");
369 				}
370 			}
371 	
372 			
373 		}
374 		
375 		
376 		/**
377 		 * Decorates the button element.  This assigns all the CSS classes
378 		 * and event handling.
379 		 * @param {HTMLElement} el The HTML element that we are styling.
380 		 * @returns {Button} Self for chaining.
381 		 */
382 		function decorateElement(el){
383 			this.getSuper(Component).decorateElement(el);
384 			var self = this;
385 
386 			$(el)
387 				
388 				.click(function(){
389 					//alert('here');
390 					self.trigger('click');
391 				})
392 				.hover(setButtonElementHover,clearButtonElementHover)
393 				.mousedown(buttonMouseDown)
394 				.mouseup(buttonMouseUp)
395 				.attr('type', 'submit')
396 				.addClass('ui-state-default')
397 				.addClass('ui-corner-all')
398 				.addClass('xatajax-button')
399 				.addClass('xatajax-button-icon-'+iconStyle)
400 				;
401 				
402 				
403 			return this;
404 				
405 		}
406 		
407 	}
408 })();
