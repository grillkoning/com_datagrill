362 		/**
363 		 * Adds a child component to this component.  This method has an optional 
364 		 * second parameter that can either be an integer or a string.  If 
365 		 * a string is provided it is treated as a label for this child so that
366 		 * it can be looked up by name.  If it is an integer, it merely
367 		 * dictates the position in the children array where the component will be placed.
368 		 * If this is not specified it is given no name, and the component will simply
369 		 * be appended to the end of the children array.
370 		 *
371 		 * @event beforeChildAdded(ComponentEvent)
372 		 * @event childAdded(ComponentEvent)
373 		 *
374 		 * @param {Component} c The child component to be added.
375 		 * @param {mixed} param  Either a string or an integer.
376 		 *
377 		 * @throws {XataJax.Exception(code: XataJax.errorcodes.COMPONENT_INDEX_OUT_OF_BOUNDS)}
378 		 *		if the specified parameter is out of bounds.
379 		 *
380 		 */
381 		function add(c, param){
382 			if ( !this.isChildAllowed(c, param) ){
383 				throw new Exception({
384 					message: 'Child component is not allowed at this location.',
385 					code: XataJax.errorcodes.CHILD_COMPONENT_NOT_ALLOWED
386 				});
387 			}
388 			var index = members.children.length;
389 			if ( typeof(param) == 'number' ){
390 				index = param;
391 			}
392 			
393 			if ( index < 0 || index > members.children.length ){
394 				throw new XataJax.Exception({
395 					message: 'Component index out of bounds.  Attempt to add component to index '+index+' failed because that is out of bounds.',
396 					code: XataJax.errorcodes.COMPONENT_INDEX_OUT_OF_BOUNDS
397 				});
398 			}
399 			
400 			
401 			
402 			var name = null;
403 			if ( typeof(param) =='string' ){
404 				name = param;
405 			}
406 			
407 			// No special parameter... we add this component to 
408 			// the end of the child list
409 			fireBeforeChildAdded(new ComponentEvent({
410 				source: this,
411 				component: c,
412 				index: index,
413 				name: name
414 			}));
415 			
416 			if ( index < members.children.length ){
417 				// we are adding this component at an index
418 				// that is already occupied by another child
419 				// component.  We need to first remove this
420 				// other component.
421 				remove(members.children[param], false);
422 				
423 			}
424 			
425 			// Check if the parent of this component is not null.
426 			if ( c.getParent() != null ){
427 				c.getParent().remove(c);
428 			}
429 			
430 			members.children.push(c);
431 			members.childIndex[c.getComponentID()] = c;
432 			members.nameToChild[name] = c;
433 			members.childToName[c.getComponentID()] = name;
434 			
435 			
436 			
437 			fireChildAdded(new ComponentEvent({
438 				source: this,
439 				component: c,
440 				index: index,
441 				name: name
442 				
443 			}));
444 			
445 				
446 				
447 				
448 				
449 			
450 		}

453 		/**
454 		 * Removes a child component.
455 		 * @param {Component} c The child component to remove.
456 		 * @param optional {boolean} repack Set false to leave an empty slot at the 
457 		 *	place where this child resides.  This is handy if you mean to replace
458 		 *	it with another component immediately.  Default value is true (i.e.
459 		 *	it repacks them.
460 		 *
461 		 * @event ComponentListener#beforeChildRemoved
462 		 * @event ComponentListener#childRemoved
463 		 *
464 		 * @throws {XataJax.Exception(code:XataJax.errorcodes.COMPONENT_NOT_FOUND)} if
465 		 *	if the specied component is not a child of this component.
466 		 */
467 		function remove(c, repack){
468 			if ( typeof(repack) == 'undefined' ){
469 				repack = true;
470 			}
471 			var index = members.children.indexOf(c);
472 			if ( index == -1 ){
473 				throw new XataJax.Exception({
474 					message: 'Failed to remove component from parent because it is not a child of this component',
475 					code: XataJax.errorcodes.COMPONENT_NOT_FOUND
476 				});
477 			}
478 			
479 			var name = members.childToName[c.getComponentID()];
480 			
481 			
482 			fireBeforeChildRemoved(new ComponentEvent({
483 				source: this,
484 				component: c,
485 				index: index,
486 				name: name
487 			}));
488 			
489 			
490 			if ( repack ){
491 				members.splice(index, 1);
492 			} else {
493 				delete members.children.indexOf(c);
494 			}
495 			
496 			if ( name ){
497 				delete members.nameToChild[name];
498 				delete members.childToName[c.getComponentID()];
499 			}
500 			delete childIndex[c.getComponentID()];
501 			
502 			fireChildRemoved(new ComponentEvent({
503 				source: this,
504 				component: c,
505 				index: index,
506 				name: name
507 			}));
508 			
509 		}

512 		/**
513 		 * Gets one or more child components from this component.
514 		 * @variant ByIndex Gets a child component by its numeric index.
515 		 * @param {int} key The index of the child to retrieve (0-based).
516 		 * @returns {Component} The child component at index <code>key</code>.
517 		 *
518 		 * @variant ByLabel Gets a child component by label.
519 		 * @param {String} key The name of the child to retrieve.
520 		 * @returns {Component} The child component stored with the given label.
521 		 *
522 		 * @variant InputObject Gets multiple child components by labels.
523 		 * @param {Object} key An object that will be used as an out parameter.  Whatever
524 		 *		keys are passed with the object are given values of the components at the
525 		 *		corresponding label within this component.
526 		 * @returns {Object} The same Object that is input as a parameter.
527 		 *
528 		 */
529 		function get(key){
530 			if ( typeof(key) == 'number' ){
531 				return members.children[key];
532 			} else if ( typeof(key) == 'string' ){
533 				return members.nameToChild[key];
534 			} else if ( typeof(key) == 'function' ){
535 				var out = [];
536 				$.each(members.children, function(){
537 					if ( key(this) ) out.push(this);
538 				});
539 			} else if ( typeof(key) == 'object' ){
540 				for ( var i in key ){
541 					key[i] = members.nameToChild[i];
542 				}
543 				return key;
544 			}
545 		}

672 		/**
673 		 * Sets or gets CSS properties for this component.  There are a few variations
674 		 * of this method:
675 		 *
676 		 * css(string key) : Returns the css value associated with string key.
677 		 * css(string key, string value) : sets the css value associated with key.
678 		 * css(void) : gets object with all current css values on this component.
679 		 * css(object values):  Sets css values base on key value pairs of values.
680 		 *
681 		 * @returns mixed
682 		 *
683 		 */
684 		function css(key, value){
685 			if ( typeof(key) == 'object' ){
686 				for ( var i in key ){
687 					this.css(i, key[i]);
688 				}
689 			} else if ( typeof(value) != 'undefined' ){
690 				if ( members.css[key] != value ){
691 					var old = members.css[key];
692 					members.css[key] = value;
693 					$(this.getElement()).css(key, value);
694 					
695 					// We fire the property change out to all
696 					// property change listeners.
697 					// Inherited from PropertyChangeSupport
698 					this.firePropertyChange('css#'+key, old, value);
699 				}
700 			} else if ( typeof(key) == 'string' ){
701 				return members.css[key];
702 			} else {
703 				return $.extend({}, members.css);
704 			}
705 		}

731 				this.firePropertyChange('classes', cls, null, idx);

822 		 * Moves a child component to a specified x, y coordinate (or the closest)
823 		 * thing.
824 		 *
825 		 * @param Component c
826 		 * @param int x
827 		 * @param int y
828 		 * @param mixed param Corresponds to the param parameter of Component#add() method
829 		 *
830 		 */
831 		function moveTo(c, x, y, param){
832 			//if ( c.
833 			var layout = this.getLayout();
834 			if ( layout != null && typeof(layout.moveTo) == 'function'){
835 				layout.moveTo(c, x, y, param);
836 			} else {
837 				var offset = $(c.getElement()).offset();
838 				var poffset = $(this.getElement()).offset();
839 				var deltaX = poffset.left + x - offset.left;
840 				var deltaY = poffset.top + y - offset.top;
841 				var leftCSS = c.css('left');
842 				var rightCSS = c.css('right');
843 				var topCSS = c.css('top');
844 				var bottomCSS = c.css('bottom');
845 				var widthCSS = c.css('width');
846 				var heightCSS = c.css('height');
847 				
848 				if ( typeof(leftCSS) != 'undefined' ){
849 					// This component explicitly specifies its left position.
850 					// So we should be able to 
851 					c.css('left', leftCSS + deltaX);
852 					
853 				}
854 				if ( typeof(rightCSS) != 'undefined' ){
855 					c.css('right', rightCSS-deltaX);
856 				}
857 				
858 				if ( typeof(topCSS) != 'undefined' ){
859 					c.css('top', topCSS+deltaY);
860 				}
861 				
862 				if( typeof(bottomCSS) != 'undefined' ){
863 					c.css('bottom', bottomCSS-deltaY);
864 				}
865 				
866 			}
867 		}

869 		/**
870 		 * Unserializes an object into this component.
871 		 *
872 		 * @param {Object} o The generic object that is being unserialized.
873 		 * @returns {Component} Self for chaining.
874 		 *
875 		 */
876 		function unserialize(o){
877 			var self = this;
878 			var data = o.data;
879 			if ( typeof(o.data) == 'object' ){
880 				if ( typeof(o.data.css) == 'object' ){
881 					$.each(o.data.css, function(key,val){
882 						self.css(key,val);
883 					});
884 				}
885 				
886 				if ( typeof(o.data.cssClasses) != 'undefined' ){
887 					$.each(o.data.cssClasses, function(){
888 						self.addClass(this);
889 					});
890 				}
891 				
892 				self.value = o.value;
893 				
894 				if ( typeof(o.data.children) != 'undefined' ){
895 					$.each(o.data.children, function(){
896 						var label = this.label;
897 						
898 						var cls = XataJax.findConstructor(this.type);
899 						if ( !cls ){
900 							throw new Exception({
901 								message: 'Could not find constructor for type '+this.type+'. Please make sure it is loaded and in the class path.'
902 							});
903 						}
904 						
905 						var obj = new cls;
906 						obj.unserialize(this);
907 						self.add(obj, label);
908 					});
909 				}
910 			}
911 			return self;
912 		
913 		}
