 32 	 * Base class for sets of records.  This class should be extended by more specific
 33 	 * classes for different types of schemas.  The load() method can be overridden
 34 	 * to load records from the datasource.
 35 	 *
 36 	 * @constructor
 37 	 *
 38 	 * @event {RecordSetEvent} beforeLoad  Fired before records are loaded into set.
 39 	 * @event {RecordSetEvent} afterLoad Fired after records are loaded into set.
 40 	 * @event {RecordSetEvent} beforeUpdate Fired before update.
 41 	 * @event {RecordSetEvent} afterUpdate Fired after update.
 42 	 * @event {RecordSetEvent} beforeRemoveRecord Fired before a record is removed.
 43 	 * @event {RecordSetEvent} afterRemoveRecord Fired after a record is removed.
 44 	 */
 45 	function RecordSet(o){
 46 	
 47 		var self = this;
 48 	
 49 		/**
 50 		 * @type {DataSource}
 51 		 */
 52 		var dataSource = null;

 88 		 * @returns {int} The total number of records in this result set (that could be loaded

105 			}
106 			return this;
107 		}

111 		/**
112 		 * Loads a segment of this record set.  This method should be overridden by 
113 		 * subclasses to actually perform the load.  The default implementation
114 		 * just sets the cardinality to 0 and all start/end values to -1.
115 		 *
116 		 * @param {int} start The index of the first record that should be loaded. (0-based).
117 		 * @param {int} end The index of the last record to be loaded.  (0-based).
118 		 * @returns {RecordSet} Self for chaining.
119 		 */
120 		function load(start, end, callback){
121 			var self = this;
122 			var req = new RecordRequest();
123 			req.setAction('load');
124 			req.setRecordSet(this);
125 			req.setParameter('start', start);
126 			req.setParameter('end', end);
127 			req.bind('complete', function(response){
128 				if ( response.getCode() == RecordResponse.SUCCESS || response.getCode() == RecordResponse.ALL_SUCCESS){
129 					self.handleLoad(response.getRecords(), start);
130 				}
131 				if ( typeof(callback) == 'function' ){
132 					callback(response);
133 				}
134 			});
135 			this.decorateLoadRequest(start, end, req);
136 			
137 			if ( dataSource == null ){
138 				throw new Exception({
139 					message: 'Null datasource.  Cannot load record set.'
140 				});
141 			}
142 			
143 			dataSource.sendRequest(req);
144 			
145 			
146 		}

163 		/**
164 		 * Handles the loading of a set of records.
165 		 * This should be called by the load() method after the loading is 
166 		 * completed - usually inside a callback.
167 		 *
168 		 * @param {array Record} recs The array of records to load.
169 		 * @param {int} index The start index where they should be inserted.
170 		 *
171 		 * @returns {RecordSet} Self for chaining.
172 		 */
173 		function handleLoad(recs, index){
174 			if ( recs.length == 0 ) return this;
175 			
176 			var lStart = index;
177 			var lEnd = index+records.length;
178 			
179 			var event = new RecordSetEvent({
180 				recordSet: this,
181 				action: 'load',
182 				records: recs,
183 				startIndex: lStart,
184 				endIndex: lEnd-1
185 				
186 			});
187 			
188 			this.trigger('beforeLoad', event);
189 			
190 			
191 			for ( var i=lStart; i<lEnd; i++){
192 				this.setRecordAt(i, recs[i-lStart]);
193 				
194 			}
195 			
196 			this.trigger('afterLoad', event);
197 			
198 			
199 			return this;
200 			
201 		}

205 		/**
206 		 * Sets the record at a specific position in this result set.  This does not imply
207 		 * that the record was added to the result set in the underlying data source - this is just a housekeeping method
208 		 * to be able to set the record at a particular index of the result set during the 
209 		 * loading of it.
210 		 *
211 		 * This fires an indexd PropertyChangeEvent for the 'records' property.
212 		 *
213 		 * @param {int} index The index within the result set that this record should be added.
214 		 * @param {Record} record The record that is being inserted.
215 		 */
216 		function setRecordAt(index, record){
217 			var self = this;
218 			if ( typeof(records[index]) == 'undefined' ){
219 				records[index] = null;
220 			}
221 			var old = records[index];
222 			if ( old != record){
223 				
224 				if ( XataJax.instanceOf(old, Record) ){
225 					this.undecorateRecord(old);
226 				}
227 				
228 				records[index] = record;
229 				this.decorateRecord(record);
230 				
231 				
232 				this.firePropertyChange('records', old, record, index);
233 			}
234 			
235 		}
236 		
237 		/**
238 		 * Decorates a record that has just been added to this record set.  This 
239 		 * adds the necessary listeners to be able to respond when the record is
240 		 * deleted (so that it can be removed from the record set) etc..
241 		 *
242 		 * @param {Record} record The record that is to be decorated.
243 		 * @returns {RecordSet} Self for chaining.
244 		 */
245 		function decorateRecord(record){
246 			var self = this;
247 			record.bind('afterDelete.updateRecordSet', function(event){
248 				self.removeRecord(record);
249 			});
250 			return this;
251 	
252 		}

255 		/**
256 		 * Strips listeners that were added to records by the decorateRecord method.
257 		 * This is generally called on records as they are being removed from the record set
258 		 * so that anything that was added to the record when it was added to the record
259 		 * set will now be removed.  This is the inverse of decorateRecord()
260 		 *
261 		 * @param {Record} record The record to undecorate.
262 		 * @returns {RecordSet} Self for chaining.
263 		 */
264 		function undecorateRecord(record){
265 			record.unbind('afterDelete.updateRecordSet');
266 		}

268 		/**
269 		 * Removes a record from the result set.  This will splice the record so that there is no blank
270 		 * hole remaining.. it just collapses all of the records up one row.  This is the correct 
271 		 * functionality to mirror a deletion of the record from the underlying datasource.  It is 
272 		 * important to understand that this action is generally taken in response to a record being
273 		 * deleted rather than to initiate a delete.  If you wish to delete a record, it is 
274 		 * most correct to call delete() on that particular record.  This method will be called
275 		 * by the afterDelete event handler for that record to update the state of the 
276 		 * result set.
277 		 *
278 		 * This will fire the beforeRemoveRecord and afterRemoveRecord events.
279 		 *
280 		 * @param {Record} record The record to remove.
281 		 * @returns {RecordSet} Self for chaining.
282 		 */
283 		function removeRecord(record){
284 			var idx = records.indexOf(record);
285 			if ( idx != -1 ){
286 				var event = new RecordSetEvent({
287 					recordSet: this,
288 					action: 'removeRecord',
289 					record: record,
290 					index: idx
291 				});
292 				this.trigger('beforeRemoveRecord', event);
293 				records.splice(idx,1);
294 				this.undecorateRecord(record);
295 				this.trigger('afterRemoveRecord', event); 
296 			}
297 			return this;
298 		}
299 		
300 		/**
301 		 * Gets the record at the specified index of the record set.  Note that this is 
302 		 * the absolute index, not the index within what's been loaded.
303 		 * @param {int} index The index to load.
304 		 *
305 		 * @returns {Record} The record at the specified index.
306 		 */
307 		function getRecordAt(index){
308 			if ( index < 0 || index > cardinality ){
309 				throw new Exception({
310 					message: 'Cannot get record at index '+index+' because that index has not been loaded yet - nor has the load been requested.'
311 				});
312 			}
313 			
314 			
315 			if ( this.indexIsLoaded(index) ){
316 				throw new Exception({
317 					message: 'Cannot get record at index '+index+' because that index has not been loaded yet.'
318 				});
319 			}
320 			
321 			return records[index];
322 			
323 		}

339 			records = [];
340 		}

346 		 *
347 		 */

349 			return (typeof(records[index]) == 'object'  && records[index] != null);
