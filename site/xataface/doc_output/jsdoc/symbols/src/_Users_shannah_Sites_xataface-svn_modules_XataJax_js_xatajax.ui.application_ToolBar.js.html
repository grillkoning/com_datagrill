1 /**
  2  * A toolbar for an application.
  3  *
  4  * This file is meant to be included in the xatjax.ui.application file
  5  * using the XataJax compiler.  It should not be referenced on its
  6  * own.
  7  * @depends xatajax.ui.application/ToolGroup.js
  8  *
  9  * @author Steve Hannah <steve@weblite.ca>
 10  * @created Feb. 5, 2011
 11  */
 12 (function(){
 13 	var $ = jQuery;
 14 	
 15 	XataJax.ui.application.ToolBar = ToolBar;
 16 	
 17 	function ToolBar(o){
 18 		
 19 		/**
 20 		 * Publish the public api for this object.
 21 		 */
 22 		$.extend(this, {
 23 			getToolGroup: getToolGroup,
 24 			setToolGroup: setToolGroup,
 25 			getElement: getElement
 26 		});
 27 	
 28 		/**
 29 		 * private member variables.
 30 		 */
 31 		var members = {
 32 			toolGroup: null,
 33 			el: null
 34 		};
 35 		
 36 		setToolGroup(new XataJax.ui.application.ToolGroup());
 37 		
 38 		
 39 		/**
 40 		 * @returns XataJax.ui.application.ToolGroup
 41 		 */
 42 		function getToolGroup(){
 43 			return members.toolGroup;
 44 		}
 45 		
 46 		/**
 47 		 * @param XataJax.ui.application.ToolGroup group
 48 		 */
 49 		function setToolGroup(group){
 50 			if ( members.toolGroup != null ){
 51 				$(members.toolGroup.getElement()).remove();
 52 			}
 53 			members.toolGroup = group;
 54 			getToolGroupWrapper().appendChild(group.getElement());
 55 		}
 56 		
 57 		/**
 58 		 * Private convenience function to retrieve the DOM element wrapper for the
 59 		 * toolgroup.
 60 		 * @returns DOMElement
 61 		 */
 62 		function getToolGroupWrapper(){
 63 			return $('.xatajax-application-toolbar-toolgroupwrapper', getElement()).get(0);
 64 		}
 65 		
 66 		/**
 67 		 * @returns DOMElement
 68 		 */
 69 		function getElement(){
 70 			if ( members.el == null ){
 71 				members.el = document.createElement('div');
 72 				$(members.el).addClass('xatajax-application-toolbar');
 73 				
 74 				var toolGroupWrapper = document.createElement('div');
 75 				$(toolGroupWrapper).addClass('xatajax-application-toolbar-toolgroupwrapper');
 76 				members.el.appendChild(toolGroupWrapper);
 77 				
 78 			}
 79 			return members.el;
 80 		}
 81 		
 82 		
 83 		
 84 	}
 85 	
 86 })();
