1 //require <xatajax.beans.core.js>
  2 (function(){
  3 	var $ = jQuery;
  4 
  5 	XataJax.beans.Subscribable = Subscribable;
  6 	
  7 	/**
  8 	 * A mixin class that adds support for events in an object.
  9 	 *
 10 	 * @constructor
 11 	 */
 12 	function Subscribable(o){
 13 	
 14 		var listeners = {};
 15 		var taggedListeners = {};
 16 		
 17 		XataJax.publicAPI(this, {
 18 			unbind: unbind,
 19 			bind: bind,
 20 			trigger: trigger
 21 		});
 22 		
 23 		
 24 		function getTag(name){
 25 			return name;
 26 		}
 27 		
 28 		function getName(name){
 29 			if ( name.indexOf('.') != -1 ){
 30 				var parts = name.split('.');
 31 				return parts[0];
 32 			} else {
 33 				return name;
 34 			}
 35 		}
 36 		
 37 		function getListenersForTag(tag){
 38 			if ( typeof(taggedListeners[tag]) == 'undefined' ){
 39 				return [];
 40 			} else {
 41 				return taggedListeners[tag];
 42 			}
 43 		}
 44 		
 45 		function addListenerForTag(tag, listener){
 46 			if ( typeof(taggedListeners[tag]) == 'undefined' ){
 47 				taggedListeners[tag] = [];
 48 			}
 49 			
 50 			taggedListeners[tag].push(listener);
 51 		}
 52 	
 53 	
 54 		/**
 55 		 * Unbinds a function from the given event.
 56 		 * @param {String} name The name of the event to unbind from.
 57 		 * @param {Function} func The function to bind to the event.
 58 		 * @returns {Subscribable} Self for chaining.
 59 		 */
 60 		function unbind(name, func){
 61 			var self = this;
 62 			var tag = getTag(name);
 63 			var name = getName(name);
 64 			if ( typeof(func) == 'undefined' ){
 65 				$.each(getListenersForTag(tag), function(){
 66 					self.unbind(name, this);
 67 				});
 68 			}
 69 			if ( typeof(listeners[name]) == 'undefined' ){
 70 				listeners[name] = [];
 71 			}
 72 			var i = listeners[name].indexOf(func);
 73 			if ( i != -1 ){
 74 				listeners[name].splice(i,1);
 75 			}
 76 			return this;
 77 		}
 78 		
 79 		/**
 80 		 * Binds a function for a given event type.  Functions will be
 81 		 * called in the context of the Subscribable object.
 82 		 *
 83 		 * @param {String} name The name of the event to bind.
 84 		 * @param {Function} func The function to bind to the event.
 85 		 * @returns {Subscribable} Self for chaining.
 86 		 */
 87 		function bind(name, func){
 88 			var tag = getTag(name);
 89 			var name = getName(name);
 90 			
 91 			addListenerForTag(tag, func);
 92 			if ( typeof(listeners[name]) == 'undefined' ){
 93 				listeners[name] = [];
 94 			}
 95 			var i = listeners[name].indexOf(func);
 96 			if ( i == -1 ){
 97 				listeners[name].push(func);
 98 			}
 99 			return this;
100 		}
101 		
102 		
103 		/**
104 		 * Triggers an event so that all functions listening to the event will be called.
105 		 *
106 		 * @param {String} name The name of the event that is being fired.
107 		 * @param {Object} event The event object.  This event can be any type of 
108 		 *		object that the Subscribable chooses to call.
109 		 * @returns {Subscribable} Self for chaining.
110 		 */
111 		function trigger(name, event){
112 			var self = this;
113 			if ( typeof(listeners[name]) == 'undefined' ){
114 				return this;
115 			}
116 			$.each(listeners[name], function(){
117 				($.proxy(this, self))(event);
118 			});
119 			
120 			return this;
121 		}
122 	}
123 })();
