 65 	/**
 66 	 * @function
 67 	 * @memberOf XataJax.util
 68 	 * @description Gets the query parameters for the current request.
 69 	 * @returns {Object} The key-value pairs of GET variables.
 70 	 *
 71 	 */
 72 	function getRequestParams(url){
 73 		
 74 		var search = window.location.search;
 75 		if ( typeof(url) != 'undefined' ){
 76 			var pos = url.indexOf('?');
 77 			if ( pos >= 0 ){
 78 				search = url.substr(pos);
 79 			} else {
 80 				search = '';
 81 			}
 82 		}
 83 		
 84 		if ( search.indexOf('?') == 0 ){
 85 			search = search.substr(1);
 86 		}
 87 		var params = search.split('&');
 88 
 89 		var out = {};
 90 
 91 		$.each(params, function(){
 92 			var parts = this.split('=');
 93 			out[decodeURIComponent(parts[0]).replace(/\+/g,' ')] = decodeURIComponent(parts[1]).replace(/\+/g,' ');
 94 		});
 95 		
 96 		return out;
 97 		
 98 	
 99 	}
100 	
101 	/**
102 	 * @function
103 	 * @memberOf XataJax.util
104 	 * @description Converts the specified query parameters into a URL with proper 
105 	 * url encoding.
106 	 * @param {Object} params Key-value pairs of GET parameters to include.
107 	 * @returns {String} A URL
108 	 */
109 	function url(params){
110 		var out = [];
111 		$.each(params, function(key,val){
112 			out.push(encodeURIComponent(key)+'='+encodeURIComponent(val));
113 		});
114 		return getSiteHref()+'?'+out.join('&')+window.location.hash;
115 	}

118 	 * @function
119 	 * @memberOf XataJax.util
120 	 * @description Dynamically loads a script by the script's path.
121 	 *
122 	 * Note that only scripts located in the Xataface Javascript include paths
123 	 * can be loaded with this method.  
124 	 
125 	 * @see <a href="http://xataface.com/dox/core/latest/class_dataface___javascript_tool.html">Dataface_JavascriptTool</a> for information
126 	 * about javascript include paths and how to set them up.
127 	 * @see <a href="http://xataface.com/dox/core/latest/classdataface__actions__load__script.html">The Xataface load_script action</a> for information on how to manipulate
128 	 * the include path on a per request basis. In particular you may find it useful to implement a handler for the beforeLoadScript event (@see The registerEventListener() method in <a href="http://xataface.com/dox/core/latest/class_dataface___application.html">Dataface_Application</a>
129 	 * @param {String} path The path to the script to load.  This should be the 
130 	 *  same format as the require directive takes.
131 	 * @param {Function} callback A callback function to be called after the script
132 	 * 	has been loaded. 
133 	 * @example
134 	 * //require <xatajax.util.js>
135 	 * var loadScript = XataJax.load('XataJax.util.loadScript');
136 	 * loadScript('xataface/IO.js', function(){
137 	 *     // Now IO.js should be loaded.
138 	 *     var IO = XataJax.load('xataface.IO');
139 	 *     IO.load('people?person_id=1', function(res){
140 	 *         alert('First Name: '+res.first_name);
141 	 *     });
142 	 * });
143 	 */
144 	function loadScript(path, callback){
145 		$.getScript(DATAFACE_SITE_HREF+'?-action=load_script&--script='+encodeURIComponent(path),
146 			callback
147 		);
148 		
149 	}
