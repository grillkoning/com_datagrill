 48 	 * Represents a single record of a schema.  Generally a record constitutes a single row in a 
 49 	 * table, but it is more abstract than that, as it could represent a tag in an XML file also.
 50 	 *
 51 	 * Records may contain fields and subrecords.  Subrecords would refer to related records 
 52 	 * in terms of a relational database, or child tags in terms of an xml file.
 53 	 *
 54 	 * @constructor
 55 	 *
 56 	 * @event {PropertyChangeEvent} beforeSetValue Event fired before a value is set.
 57 	 * @event {PropertyChangeEvent} afterSetValue Event fired after a value is set.
 58 	 * @event {RecordResponse} afterCommit Fired after commit is successfully completed.
 59 	 * @event {RecordResponse} failedCommit Fired after a commit attempt has failed.
 60 	 * @event {RecordEvent} beforeCommit Fired just before a commit is attempted.
 61 	 * @event {RecordEvent} beforeRollback Fired just before a rollback is performed.
 62 	 * @event {RecordEvent} afterRollback Fired just after a rollback is performed.

152 		 * Sets the schema for this record.
153 		 *
154 		 * @param {Schema} s The schema.
155 		 * @returns {Record} Self for chaining.
156 		 */
157 		function setSchema(s){
158 			if ( s != schema ){
159 				var old = schema;
160 				if ( XataJax.instanceOf(old, Schema) ){
161 					old.undecorateRecord(this);
162 				}
163 				schema = s;
164 				if ( XataJax.instanceOf(schema, Schema) ){
165 					schema.decorateRecord(this);
166 				}
167 				this.firePropertyChange('schema', old, s);
168 			}
169 			return this;
170 		}

252 		/**
253 		 * Returns the value of this record in the specified field (i.e. the field
254 		 * of this record such that its path = path
255 		 *
256 		 * @param {String} path The path to the field whose value we wish to retrieve.
257 		 * @returns {mixed} The value of the field in this record.
258 		 * @throws {Exception(code: XataJax.errorcodes.PATH_NOT_IN_SCHEMA)} If the schema does not 
259 		 *		contain a field at this path.
260 		 */
261 		function getValue(path){
262 		
263 			
264 			if ( !schema.containsField(path) ){
265 				throw new Exception({
266 					message: 'Attempt to set value in field at path '+path+' but the schema contains no such field.',
267 					code: XataJax.errorcodes.PATH_NOT_IN_SCHEMA
268 				});
269 			}
270 			return values[path];

274 		/**
275 		 * Sets the value of a field in this record.
276 		 *
277 		 * @param {String} path The path to the field whose value we are setting.
278 		 * @param {mixed} value The value to set for this field.
279 		 * @throws {Exception(code: XataJax.errorcodes.PATH_NOT_IN_SCHEMA)} If the schema doesn't 
280 		 *		contain a field at this path.
281 		 *
282 		 * @returns {Record} Self for chaining.
283 		 */
284 		function setValue(path, value){
285 			if ( snapshot == null ){
286 				makeSnapshot();
287 			}
288 			if ( !schema.containsField(path) ){
289 				throw new Exception({
290 					message: 'Attempt to set value in field at path '+path+' but the schema contains no such field.',
291 					code: XataJax.errorcodes.PATH_NOT_IN_SCHEMA
292 				});
293 			}
294 			var old = this.getValue(path);
295 			value = this.prepareValue(path, value);
296 			if ( old != value){
297 				var event = new PropertyChangeEvent({
298 					source: this,
299 					propertyName: path,
300 					oldValue: old,
301 					newValue: value
302 				});
303 				this.trigger('beforeSetValue',event);
304 				
305 				
306 				
307 				values[path] = value;
308 				dirty = true;
309 				
310 				this.trigger('afterSetValue', event);
311 				
312 				
313 				
314 				
315 			}
316 			
317 			return this;
318 		}

336 		 * Gets multiple values at once.
337 		 *
338 		 * @variant 1 Returns object with the values.  Object keys are the field paths, and their corresponding
339 		 *		values are the values.
340 		 * @returns {Object} The output values.  Contains all fields in schema.
341 		 * 
342 		 * @variant 2 Populates an input object with values for the keys provided in the object.
343 		 * @param {Object} outValues The object with keys to be populated.  This same object's values
344 		 * 	will be populated by this method.
345 		 * @returns {Object} Another reference to outValues so that this method can be called the same
346 		 * 	way as for the no-argument version.
347 		 */
348 		function getValues(outValues){
349 			var self = this;
350 			if ( typeof(outValues) == 'object' ){
351 				$.each(outValues, function(key,value){
352 					outValues[key] = self.getValue(key);
353 				});
354 				return values;
355 			} else {
356 				outValues = {};
357 				$.each(schema.getFields(), function(key, value){
358 					outValues[key] = self.getValue(key);
359 				});
360 				return outValues;
361 			}
362 		}

378 		/**
379 		 * <p>Gets the subrecord set (corresponding to a subschema of the record's schema
380 		 * at the specified path.  This does not recursively search the subschemas.  We'll 
381 		 * want to make recursive access a distinct action because it is possible
382 		 * that we would need to load extra information from the server in order to 
383 		 * perform such a request.  This is because the sub-sub schemas will depend
384 		 * on which record of the sub-schema it branches from.</p>
385 		 *
386 		 * <p>Note that the record set that is returned won't contain any records until 
387 		 * they are loaded explicitly.</p>
388 		 *
389 		 * <p>Note 2: The if this is the first time that record set has been loaded,
390 		 *  the recordset will be created by the createSubRecordSet(path) method.  Subclasses
391 		 *  would probably override that method to be able to correctly produce the subrecord
392 		 * 	sets according to the data type.
393 		 * </p>
394 		 *
395 		 * @param {String} path The path to the subschema whose record set we are retrieving.
396 		 * @returns {RecordSet} The record set corresponding with that schema.
397 		 * @throws {Exception} If there is no subschema at that path.
398 		 */
399 		function getSubRecordSet(path){
400 			if ( !schema.containsSubSchema(path) ){
401 				throw new Exception({
402 					message: 'The schema does not contain a subschema at path '+path+' so you can not retrieve subrecords at that path.'
403 				});
404 			}
405 			
406 			if ( typeof(subRecords[path]) == 'undefined' ){
407 				 subRecords[path] = this.createSubRecordSet(path);
408 			}
409 			
410 			return subRecords[path];
411 		}

485 		/**
486 		 * Commits any changes to the datasource.
487 		 *
488 		 * @param {Function} callback A function to be called after the commit is complete.
489 		 * 	it should be passed a RecordCommitResponseEvent so that it knows what happened
490 		 *  with the commit.
491 		 * @param {boolean} deep If true then this will try to commit any changes to subrecord sets
492 		 *  also.
493 		 * @returns {Record} Self for chaining.
494 		 */
495 		function commit(callback, deep){
496 			var self = this;
497 			if ( typeof(deep) == 'undefined' ) deep = false;
498 			
499 			if ( dataSource == null ){
500 				throw new Exception({
501 					message: 'Cannot commit record because the data source is currently null.',
502 					code: XataJax.errorcodes.NOT_BOUND_TO_DATASOURCE
503 				});
504 			}
505 			
506 			if ( this.isDirty() ){
507 				var event = new RecordEvent({
508 					record: this,
509 					action: 'commit'
510 				});
511 				
512 				if ( this.isNew() || this.getStatus() == Record.STATUS_EXISTING ){
513 					this.trigger('beforeSave', new RecordEvent({
514 						record: this,
515 						action: 'save'
516 					}));
517 				}
518 				
519 				if ( this.isNew() ){
520 					this.trigger('beforeInsert', new RecordEvent({
521 						record: this,
522 						action: 'insert'
523 					}));
524 					event.action = 'insert';
525 				} else if ( this.isCondemned() ){
526 					this.trigger('beforeDelete', new RecordEvent({
527 						record: this,
528 						action: 'delete'
529 					}));
530 					event.action = 'delete';
531 				} else if ( this.getStatus() == Record.STATUS_EXISTING) {
532 					this.trigger('beforeUpdate', new RecordEvent({
533 						record: this,
534 						action: 'update'
535 					}));
536 					event.action = 'update';
537 				} else {
538 					throw new Exception({
539 						message: 'Cannot commit record because it has an invalid status: '+this.getStatus()+'.  Status must be one of Condemned, Existing, or New.',
540 						code: XataJax.errorcodes.INVALID_RECORD_STATUS
541 					});
542 				}
543 				this.trigger('beforeCommit', event);
544 				
545 				var request = new RecordRequest();
546 				request.bind('complete', function(e){self.handleCommitResponse(e);});
547 				request.bind('complete', callback);
548 				request.addRecord(this);
549 				request.setAction('commit');
550 				
551 				dataSource.sendRequest(request);
552 			}
553 			
554 			//trigger('afterCommit', event);
555 			return this;
556 		}
557 		
558 		/**
559 		 * @param {RecordResponse}
560 		 */
561 		function handleCommitResponse(response){
562 			var singleResponse = response.getResponseForRecord(this);
563 			if ( singleResponse.getCode() == RecordResponse.SUCCESS ){
564 				this.setDirty(false);
565 				makeSnapshot();
566 				
567 				
568 				
569 				this.trigger('afterCommit', response);
570 				var afterSave = false;
571 				if ( this.isNew() ){
572 					afterSave = true;
573 					this.setStatus(Record.STATUS_EXISTING);
574 					this.trigger('afterInsert', new RecordEvent({
575 						record: this,
576 						action: 'insert'
577 					}));
578 					
579 				} else if ( this.isCondemned() ){
580 					this.setStatus(Record.STATUS_DELETED);
581 					this.trigger('afterDelete', new RecordEvent({
582 						record: this,
583 						action: 'delete'
584 					
585 					}));
586 					
587 				} else {
588 					afterSave = true;
589 					this.trigger('afterUpdate', new RecordEvent({
590 						record: this,
591 						action: 'update'
592 					}));
593 				
594 				}
595 				
596 				if ( afterSave ){
597 					this.trigger('afterSave', new RecordEvent({
598 						record: this,
599 						action: 'save'
600 					}));
601 				}
602 				
603 			} else {
604 				this.trigger('failedCommit', response);
605 			}
606 		}

611 		 * Checks if the value in a field is changed.
612 		 * @param {String} path The path to the field that we are checking.
613 		 * @returns {boolean}
614 		 */
615 		function valueChanged(path){
616 			if ( snapshot == null ){
617 				makeSnapshot();
618 			}
619 			if ( !schema.containsField(path) ){
620 				throw new Exception({
621 					message: 'Attempt to check status of value in field at path '+path+' but the schema contains no such field.',
622 					code: XataJax.errorcodes.PATH_NOT_IN_SCHEMA
623 				});
624 			}
625 			
626 			return (snapshot[path] != values[path]);
627 		}
