 28 	/**
 29 	 * Returns the DOMElement that represents the slot that can
 30 	 * be added to in the ancestor chain for a given element target.  This is 
 31 	 * helpful when testing where to drop a component in the designer
 32 	 * so that we know which components can actually be dropped onto.
 33 	 *
 34 	 * @param DOMElement el The child element of the slot we are looking for.  This
 35 	 * 	will usually be the event.target of a jquery event.
 36 	 *
 37 	 * @param int index The index of the element.  Default is 0.  If this number
 38 	 * 	is greater than 0 then it will use a slot higher up in the chain.
 39 	 *
 40 	 * @param function indicator An indicator function that can optionally provided
 41 	 * 	to indicate whether a given slot is eligible.  This is handy if a component
 42 	 *	only allows certain types of components to be dropped in some slots.  Then
 43 	 *	they can provide this parameter as a function to return true.
 44 	 *
 45 	 * @returns DOMElement The DOM Element where a component can be added.
 46 	 */
 47 	function getSlot(el, index, indicator){
 48 		if ( typeof(indicator) != 'function' ) indicator = function(){return true;}
 49 		if ( typeof(index) == 'undefined' ){
 50 			index = 0;
 51 		}
 52 		var currEl = $(el);
 53 		while ( currEl.size() > 0 ){
 54 			var slot = currEl.parent('.xatajax-ui-component-slot');
 55 			if ( indicator(slot) && index--==0 ) return $(slot).get(0);
 56 			else currEl = slot;
 57 		}
 58 		return null;
 59 		
 60 	}

 63 	/**
 64 	 * Gets the component where a specified child component can be dropped.
 65 	 *
 66 	 * @param Component component The child component that we are testing
 67 	 * 	to try to drop.
 68 	 *
 69 	 * @param DOMElement targetEl The target of the mouse event.  This would
 70 	 *	be a descendent of the drop target (if one is found).
 71 	 *
 72 	 * @param int index If we don't want to take the first matching drop
 73 	 * 	target slot, this optional integer can be used to indicate which
 74 	 *	one we want to take.  E.g. index=1 would indicate that we want
 75 	 *	to take the second ancestor we find in the component heirarchy. Default
 76 	 *	value is 0.
 77 	 *
 78 	 * @returns object(component: COMPONENT, param: SLOTNAME or INDEX) An object with the
 79 	 *		parent component to which to add the child, and the name of the slot
 80 	 *		(if any) in which to add the child to said component.
 81 	 *
 82 	 */
 83 	function getDropTarget(component, targetEl, index){
 84 		return getSlot(targetEl, index, function(el){
 85 			var wrapperComponent = Component.getComponentWrapper(el);
 86 			var slotName = $(el).data('xatajax-component-slot');
 87 			
 88 			if ( wrapperComponent != null && wrapperComponent.isChildAllowed(component, slotName) ){
 89 				return true;
 90 			}
 91 			return false;
 92 		});
 93 		
 94 			
 95 	}

 97 	function getDropComponentTarget(component, targetEl, index){
 98 		var slot = getDropTarget(component, targetEl, index);
 99 		
100 		if ( !slot ) return null;
101 		
102 		var wrapperComponent = Component.getComponentWrapper(el);
103 		var slotName = $(el).data('xatajax-component-slot');
104 		if ( !wrapperComponent ){
105 			throw new Exception({
106 				message: 'Internal error trying to find wrapper component for drop target.'
107 			});
108 		}
109 		
110 		
111 		return {
112 			component: wrapperComponent,
113 			param: slotName
114 		};
115 	}

651 		 * @event PropertyChangeEvent(selectedComponent : Component)
652 		 * @event PropertyChangeEvent(selectedComponents: array(Component))

655 			var el = $(c.getElement());
656 			var cls = 'xatajax-ui-designer-selected';
657 			if ( sel ){
658 				if ( !el.hasClass(cls) ){
659 					el.addClass(cls)
660 					
661 				}

663 				if ( idx == -1 ){
664 					var old = $.merge([], selectedComponents);
665 					selectedComponents.push(c);
666 					//alert('decorating selected');
667 					this.decorateSelected(c);
668 					this.firePropertyChange('selectedComponent', null, c, selectedComponents.length);
669 					this.firePropertyChange('selectedComponents', old, selectedComponents);
670 					
671 				}
672 			} else {
673 				if ( el.hasClass(cls) ){
674 					el.removeClass(cls);
675 					var idx = selectedComponents.indexOf(c);
676 					if ( idx != -1 ){
677 						var old = $.merge([], selectedComponents);
678 						selectedComponents.splice(idx, 1);
679 						this.undecorateSelected(c);
680 						this.firePropertyChange('selectedComponent', c, null, idx);
681 						this.firePropertyChange('selectedComponents', old, selectedComponents);
682 						
683 					}
684 				}
685 			}

786 		/**
787 		 * Repositions the drag handle for a component.  This is sometimes necessary if
788 		 * the component has been moved or resized.
789 		 *
790 		 * @param Component c The component onto which we are installing the drag handle.
791 		 */
792 		function updateDragHandle(c){
793 		
794 			if ( c.dragHandle != null ){
795 				var cOffset = $(c.getElement()).offset();
796 				var dOffset = $(self.getElement()).offset();
797 				var rOffset = {
798 					top: cOffset.top-dOffset.top,
799 					left: cOffset.left-dOffset.left
800 				};
801 				
802 				var cWidth = $(c.getElement()).width();
803 				var cHeight = $(c.getElement()).height();
804 				
805 				var hWidth = $(c.dragHandle).width();
806 				var hHeight = $(c.dragHandle).height();
807 				
808 				// We want the drag handle centered on the element
809 				$(c.dragHandle).offset({
810 					top: rOffset.top + cHeight/2 - hHeight/2,
811 					left: rOffset.left + cWidth/2 - hWidth/2
812 				});
813 			}
814 		}

818 		 * Adds a drag handle to a component.  This is generally performed with
819 		 * the component is selected so that it gives the user an opportunity to 
820 		 * reposition the component by dragging it.
821 		 *
822 		 * @param Component c The component to which the drag handle is being added.
823 		 */
824 		function addDragHandle(c){
825 			//alert("adding handle");
826 			if ( !c.dragHandle  ){
827 				//alert("in if");
828 				var handle = document.createElement('div');
829 				c.dragHandle = handle;
830 				$(handle).addClass('xatajax-designer-draghandle');
831 				$(handle).addClass('ui-icon');
832 				$(handle).addClass('ui-icon-arrow-4');
833 				$(c.getElement()).draggable({
834 					handle: handle,
835 					helper: function(){
836 						var el = c.getElement();
837 						var width = $(el).width();
838 						var height = $(el).height();
839 						var offset = $(el).offset();
840 						
841 						
842 						var silhouette = document.createElement('div');
843 						$(silhouette).addClass('xatajax-component-moving-silhouette');
844 						$(self.getElement()).append(silhouette);
845 						$(silhouette)
846 							.width(width)
847 							.height(height)
848 							.offset(offset);
849 						return silhouette;
850 					},
851 					opacity: 0.35,
852 					revert: 'invalid',
853 					stack: '.xatajax-component',
854 					appendTo: self.getElement()
855 					//containment: self.getElement()
856 				});
857 				
858 				$(c.getElement()).append(c.dragHandle);
859 				updateDragHandle(c);
860 			}
