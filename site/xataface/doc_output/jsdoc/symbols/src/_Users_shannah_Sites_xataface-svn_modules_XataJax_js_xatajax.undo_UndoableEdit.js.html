1 //require <xatajax.undo.js>
  2 (function(){
  3 	var $ = jQuery;
  4 	var Exception = XataJax.Exception;
  5 	
  6 	XataJax.undo.UndoableEdit = UndoableEdit;
  7 	
  8 	/**
  9 	 * Error code for exceptions thrown when an undo cannot be done.
 10 	 */
 11 	XataJax.errorcodes.CANNOT_UNDO_EXCEPTION = XataJax.nextErrorCode();
 12 	
 13 	/**
 14 	 * Error code for exceptions thrown when a redo cannot be done.
 15 	 */
 16 	XataJax.errorcodes.CANNOT_REDO_EXCEPTION = XataJax.nextErrorCode();
 17 	
 18 	
 19 	
 20 	/**
 21 	 * Constructor for UndoableEdit class.  This is basically a javascript port
 22 	 * of the javax.swing.undo.UndoableEdit
 23 	 *
 24 	 * @param {String} presentationName  The presentation name of the edit.
 25 	 * @param {boolean} significant Flag indicating whether edit is significant.  Default true
 26 	 * @param {Function} undo Operation to perform to undo this edit.
 27 	 * @param {Function} redo Operation to perform to redo this edit.
 28 	 *
 29 	 * @constructor
 30 	 *
 31 	 """
 32 	 new UndoableEdit();
 33 	 """
 34 	 */
 35 	function UndoableEdit(o){
 36 		/**
 37 		 """
 38 		 UndoableEdit.__methods__ = publicAPI;
 39 		 """
 40 		*/
 41 		/**
 42 		 * Extends Object
 43 		 */
 44 		 
 45 		XataJax.extend(this, new Object());
 46 		
 47 		var publicAPI = {
 48 			undo: undo,
 49 			redo: redo,
 50 			addEdit: addEdit,
 51 			die: die,
 52 			canUndo: canUndo,
 53 			canRedo: canRedo,
 54 			getPresentationName: getPresentationName,
 55 			isSignificant: isSignificant,
 56 			setSignificant: setSignificant
 57 			
 58 		};
 59 		
 60 		/**
 61 		 * Register public API methods
 62 		 */
 63 		XataJax.publicAPI(this, publicAPI);
 64 		
 65 		var _protected = {
 66 			_undo: function(){},
 67 			_redo: function(){}
 68 		};
 69 		
 70 		if ( typeof(o) == 'object' ){
 71 			if ( typeof(o.undo) == 'function' ){
 72 				_protected._undo = o.undo;
 73 			}
 74 			
 75 			if ( typeof(o.redo) == 'function' ){
 76 				_protected._redo = o.redo;
 77 			}
 78 			
 79 			if ( !typeof(this._protected)=='object' ){
 80 				this._protected = {};
 81 			}
 82 		}
 83 		
 84 		$.extend(this._protected, _protected);
 85 		
 86 		
 87 		/**
 88 		 * Private member variables
 89 		 */
 90 		
 91 		/**
 92 		 * @type {String}
 93 		 * The presentation name of the edit.
 94 		 */
 95 		var presentationName = '';
 96 		
 97 		
 98 		/**
 99 		 * @type {boolean}
100 		 * Whether this edit is significant or not.
101 		 */
102 		var significant = true;
103 		
104 		if ( typeof(o) == 'object' ){
105 			if ( typeof(o.presentationName) == 'string' ){
106 				presentationName = o.presentationName;
107 			}
108 			if ( typeof(o.significant) == 'boolean' ){
109 				significant = o.significant;
110 			}
111 		}
112 	
113 		
114 		/**
115 		 * @type {boolean}
116 		 * Indicates whether this edit is alive or not.  If the edit
117 		 * is not alive it can neither be redone nor undone.
118 		 *
119 		 */
120 		var alive = true;
121 		
122 		/**
123 		 * @type {boolean}
124 		 *
125 		 * Flag indicating whether the edit has been done or not.  If the edit 
126 		 * has been done it can be undone.  If it hasn't been done, it can be
127 		 * redone.
128 		 */
129 		var hasBeenDone = true;
130 		
131 		/**
132 		 * @type {array UndoableEdit}
133 		 *
134 		 * An array storing the child edits of this edit (the edit could comprise 
135 		 * multiple edits.
136 		 *
137 		 */
138 		var edits = [];
139 		
140 		/**
141 		 * Adds an edit to this edit.
142 		 *
143 		 * @param {UndoableEdit} edit
144 		 */
145 		function addEdit(edit){
146 			edits.push(edit);
147 			if ( edit.isSignificant() ){
148 				this.setSignificant(true);
149 			}
150 		}
151 		
152 		
153 		/**
154 		 * Returns whether this edit is significant or not.
155 		 * @returns {boolean}
156 		 */
157 		function isSignificant(){
158 			return significant;
159 		}
160 		
161 		/**
162 		 * Sets whether this edit is significant or not.
163 		 *
164 		 * @param {boolean} s
165 		 */
166 		function setSignificant(s){
167 			significant = s;
168 		}
169 		
170 		/**
171 		 * Undoes this edit.
172 		 * @throws {Exception(code=XataJax.errorcodes.CANNOT_UNDO_EXCEPTION)} if edit cannot be undone.
173 		 * @returns {void}
174 		 */
175 		function undo(){
176 			if ( this.canUndo() ){
177 				for ( var edit in edits){
178 					if ( typeof(edit.undo) == 'function' ){
179 						edit.undo();
180 					}
181 				}
182 				if ( typeof(this._protected._undo) == 'function' ){
183 					this._protected._undo();
184 				}
185 				hasBeenDone = false;
186 			} else {
187 				throw new Exception({
188 					message: 'Cannot undo this edit',
189 					code: XataJax.errorcodes.CANNOT_UNDO_EXCEPTION
190 				});
191 			}
192 		}
193 		
194 		/**
195 		 * Redoes this edit.
196 		 *
197 		 * @throws {Exception(code=XataJax.errorcodes.CANNOT_REDO_EXCEPTION)} if edit cannot be redone.
198 		 */
199 		function redo(){
200 			if ( this.canRedo() ){
201 				for ( var edit in edits ){
202 					if ( typeof(edit.redo) == 'function' ){
203 						edit.redo();
204 					}
205 				}
206 				if ( typeof(this._protected._redo) == 'function' ){
207 					this._protected._redo();
208 				}
209 				hasBeenDone = true;
210 			} else {
211 				throw new Exception({
212 					message: 'Cannot redo this edit',
213 					code: XataJax.errorcodes.CANNOT_REDO_EXCEPTION
214 				});
215 			}
216 		}
217 		
218 		/**
219 		 * Kills this edit so that it can neither be undone nor redone.
220 		 *
221 		 * @returns {void}
222 		 */
223 		function die(){
224 			alive = false;
225 		}
226 		
227 		/**
228 		 * Returns boolean telling whether this edit can be undone.  This
229 		 * returns true iff the edit is alive and it has been done.
230 		 * 
231 		 * @returns {boolean}
232 		 */
233 		function canUndo(){
234 			return (alive && hasBeenDone);
235 		}
236 		
237 		
238 		/**
239 		 * Returns a boolean telling whether this edit can be redone.  This 
240 		 * returns true iff the edit is alive and has not been done.
241 		 *
242 		 * @returns {boolean}
243 		 */
244 		function canRedo(){
245 			return (alive && !hasBeenDone);
246 		
247 		}
248 		
249 		
250 		/**
251 		 * Returns the presentation name of this edit.
252 		 * @returns {String}
253 		 */
254 		function getPresentationName(){
255 			return presentationName;
256 		}
257 		
258 		
259 		
260 		
261 	
262 	}
263 	
264 	/**
265 	 * The prototype for undoableedit which defines stub placeholders
266 	 * for methods.
267 	 */
268 	UndoableEdit.prototype = {
269 		addEdit: function(edit){
270 		
271 		},
272 		
273 		
274 		canUndo: function(){return false;},
275 		canRedo: function(){return false;},
276 		
277 		die: function(){
278 			
279 			
280 		},
281 		
282 		getPresentationName: function(){
283 			return '';
284 		},
285 		
286 		getRedoPresentationName: function(){
287 			return 'Redo '+this.getPresentationName();
288 		},
289 		
290 		getUndoPresentationName: function(){
291 		
292 			return 'Undo '+this.getPresentationName();
293 		},
294 		
295 		isSignificant: function(){ return true;},
296 		
297 		redo: function(){},
298 		
299 		replaceEdit: function(edit){
300 			return false;
301 		},
302 		
303 		undo: function(){}
304 		
305 		
306 	};
307 })();
