 92 	/**
 93 	 * Searches for a given element in the array, returns -1 if it is not present.
 94 	 * @param {Number} [from] The index at which to begin the search
 95 	 */
 96 	qq.indexOf = function(arr, elt, from){
 97 		if (arr.indexOf) return arr.indexOf(elt, from);
 98 		
 99 		from = from || 0;
100 		var len = arr.length;    
101 		
102 		if (from < 0) from += len;  
103 	
104 		for (; from < len; from++){  
105 			if (from in arr && arr[from] === elt){  
106 				return from;
107 			}
108 		}  
109 		return -1;  
110 	};

248 	/**
249 	 * obj2url() takes a json-object as argument and generates
250 	 * a querystring. pretty much like jQuery.param()
251 	 * 
252 	 * how to use:
253 	 *
254 	 *    `qq.obj2url({a:'b',c:'d'},'http://any.url/upload?otherParam=value');`
255 	 *
256 	 * will result in:
257 	 *
258 	 *    `http://any.url/upload?otherParam=value&a=b&c=d`
259 	 *
260 	 * @param  Object JSON-Object
261 	 * @param  String current querystring-part
262 	 * @return String encoded querystring
263 	 */
264 	qq.obj2url = function(obj, temp, prefixDone){
265 		var uristrings = [],
266 			prefix = '&',
267 			add = function(nextObj, i){
268 				var nextTemp = temp 
269 					? (/\[\]$/.test(temp)) // prevent double-encoding
270 					   ? temp
271 					   : temp+'['+i+']'
272 					: i;
273 				if ((nextTemp != 'undefined') && (i != 'undefined')) {  
274 					uristrings.push(
275 						(typeof nextObj === 'object') 
276 							? qq.obj2url(nextObj, nextTemp, true)
277 							: (Object.prototype.toString.call(nextObj) === '[object Function]')
278 								? encodeURIComponent(nextTemp) + '=' + encodeURIComponent(nextObj())
279 								: encodeURIComponent(nextTemp) + '=' + encodeURIComponent(nextObj)                                                          
280 					);
281 				}
282 			}; 
283 	
284 		if (!prefixDone && temp) {
285 		  prefix = (/\?/.test(temp)) ? (/\?$/.test(temp)) ? '' : '&' : '?';
286 		  uristrings.push(temp);
287 		  uristrings.push(qq.obj2url(obj));
288 		} else if ((Object.prototype.toString.call(obj) === '[object Array]') && (typeof obj != 'undefined') ) {
289 			// we wont use a for-in-loop on an array (performance)
290 			for (var i = 0, len = obj.length; i < len; ++i){
291 				add(obj[i], i);
292 			}
293 		} else if ((typeof obj != 'undefined') && (obj !== null) && (typeof obj === "object")){
294 			// for anything else but a scalar, we will use for-in-loop
295 			for (var i in obj){
296 				add(obj[i], i);
297 			}
298 		} else {
299 			uristrings.push(encodeURIComponent(temp) + '=' + encodeURIComponent(obj));
300 		}
301 	
302 		return uristrings.join(prefix)
303 						 .replace(/^&/, '')
304 						 .replace(/%20/g, '+'); 
305 	};

467 		_validateFile: function(file){
468 			var name, size;
469 			
470 			if (file.value){
471 				// it is a file input            
472 				// get input value and remove path to normalize
473 				name = file.value.replace(/.*(\/|\\)/, "");
474 			} else {
475 				// fix missing properties in Safari
476 				name = file.fileName != null ? file.fileName : file.name;
477 				size = file.fileSize != null ? file.fileSize : file.size;
478 			}
479 						
480 			if (! this._isAllowedExtension(name)){            
481 				this._error('typeError', name);
482 				return false;
483 				
484 			} else if (size === 0){            
485 				this._error('emptyError', name);
486 				return false;
487 														 
488 			} else if (size && this._options.sizeLimit && size > this._options.sizeLimit){            
489 				this._error('sizeError', name);
490 				return false;
491 							
492 			} else if (size && size < this._options.minSizeLimit){
493 				this._error('minSizeError', name);
494 				return false;            
495 			}
496 			
497 			return true;                
498 		},

500 			var message = this._options.messages[code];        
501 			function r(name, replacement){ message = message.replace(name, replacement); }
502 			
503 			r('{file}', this._formatFileName(fileName));        
504 			r('{extensions}', this._options.allowedExtensions.join(', '));
505 			r('{sizeLimit}', this._formatSize(this._options.sizeLimit));
506 			r('{minSizeLimit}', this._formatSize(this._options.minSizeLimit));
507 			
508 			this._options.showMessage(message);                
509 		},

517 			var ext = (-1 !== fileName.indexOf('.')) ? fileName.replace(/.*[.]/, '').toLowerCase() : '';
518 			var allowed = this._options.allowedExtensions;
519 			
520 			if (!allowed.length){return true;}        
521 			
522 			for (var i=0; i<allowed.length; i++){
523 				if (allowed[i].toLowerCase() == ext){ return true;}    
524 			}
525 			
526 			return false;
527 		},    
528 		_formatSize: function(bytes){
529 			var i = -1;                                    
530 			do {
531 				bytes = bytes / 1024;
532 				i++;  
533 			} while (bytes > 99);
534 			
535 			return Math.max(bytes, 0.1).toFixed(1) + ['kB', 'MB', 'GB', 'TB', 'PB', 'EB'][i];          
536 		}
