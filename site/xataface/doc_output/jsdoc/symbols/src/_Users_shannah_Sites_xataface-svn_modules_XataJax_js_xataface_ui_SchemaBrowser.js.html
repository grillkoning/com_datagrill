 46 	/**
 47 	 * The SchemaBrowser is a simple component that allows users to browse the 
 48 	 * schema of a table.  This includes fields (grafted, native, and calculated),
 49 	 * relationships, and related fields - and may be extended later to include
 50 	 * other aspects.
 51 	 *
 52 	 * The primary purpose of this component is for building reports in a WYSIWYG e
 53 	 * editor so that users can easily see which fields are available, and add
 54 	 * them to their reports with click or a drag.
 55 	 *
 56 	 * @constructor
 57 	 *
 58 	 * @event nodeClicked {NodeClickedEvent} Event fired when any node in the tree is clicked.
 59 	 * @event fieldClicked {FieldClickedEvent} Event fired when a field node in the tree is clicked.
 60 	 *
 61 	 *
 62 	 * Events:
 63 	 * =======
 64 	 *
 65 	 * NodeClickedEvent {
 66 	 * 		sourceEvent : <JQuery Event Object>  // The underlying jQuery event object
 67 	 *		sourceElement: <HTMLElement> 		 // The A tag that was clicked
 68 	 *		sourceBrowser: <SchemaBrowser>		 // The SchemaBrowser that originated the event.
 69 	 * }
 70 	 *
 71 	 * FieldClickedEvent {
 72 	 * 		sourceEvent : <JQuery Event Object>  // The underlying jQuery event object
 73 	 *		sourceElement: <HTMLElement> 		 // The A tag that was clicked
 74 	 *		sourceBrowser: <SchemaBrowser>		 // The SchemaBrowser that originated the event.
 75 	 *		fieldName: <String>					 // The name of the filed that was clicked.
 76 	 *		macro: <String>						 // The macro for the field (e.g. {$fieldname}
 77 	 * }
 78 	 *
 79 	 * @author Steve Hannah <steve@weblite.ca>
 80 	 * @created July 11, 2011
 81 	 */

186 	 * Initializes the tree for a schema browser.
187 	 *
188 	 * @param {SchemaBrowser} sb The schema browser for which the tree is
189 	 *  being initialized.
190 	 * @return {void}
191 	 */
192 	function initTree(sb){
193 	
194 		var q = {
195 			'-table': sb.query['-table'],
196 			'-action': 'xf_schemabrowser_getschema'
197 		
198 		};
199 		
200 		$(sb.el)
201 			.bind('loaded.jstree', function(event, data){
202 				// Now let's load the previews for it
203 				
204 				initTreePreview(sb, this);
205 				initTreeEvents(sb, this);
206 			
207 			})
208 			.jstree({
209 			'plugins': ['themes','json_data'],
210 			'json_data': {
211 				'ajax': {
212 					url: DATAFACE_SITE_HREF,
213 					data: q,
214 					success: function(res){
215 						try {
216 							if ( res.code == 200 ) return res.schema;
217 							else if ( res.message ) throw res.message;
218 							else throw 'Faild to load fields.  See server log for details.';
219 						} catch(e){
220 							alert(e);
221 						}
222 					}
223 				}
224 			}
225 		});

228 	}

230 	/**
231 	 * Initializes events on a tree.
232 	 * @param {SchemaBrowser} sb The schema browser we are initializing.
233 	 * @param {HTMLElement} el The HTML element for the tree part of the browser.
234 	 * @return {void}
235 	 */
236 	function initTreeEvents(sb, el){
237 	
238 		$('li a').click(function(event){
239 			var a = this;
240 			sb.trigger('nodeClicked', {
241 				sourceEvent: event,
242 				sourceElement: a,
243 				sourceBrowser: sb
244 			});
245 			return false;
246 		});
247 	
248 	
249 		$('li[xf-schemabrowser-macro] a').click(function(event){
250 			var a = this;
251 			sb.trigger('fieldClicked', {
252 				sourceEvent: event,
253 				sourceElement: a,
254 				sourceBrowser: sb,
255 				fieldName: $(a).parent('li').attr('xf-schemabrowser-fieldname'),
256 				macro: $(a).parent('li').attr('xf-schemabrowser-macro')
257 			});
258 			return false;
259 		});
260 	}

263 	 * The SchemaBrowser component provides a preview of the record data associated with

274 		refreshTreePreview(sb,el);

278 	/**
279 	 * Updates the preview of the record data in the tree.  This should be called each
280 	 * time some parameters are changed that might cause different record data to 
281 	 * show up.  This includes:
282 	 *
283 	 * - sb.query['-table']
284 	 * - sb.query['-cursor']
285 	 *
286 	 * @param {SchemaBrowser} sb The subject schema browser.
287 	 * @param {HTMLElement} el The tree's element.
288 	 * @return {void}
289 	 */
290 	function refreshTreePreview(sb, el){
291 	
292 		var url = DATAFACE_SITE_HREF;
293 		var q = {
294 			'-table': sb.query['-table'],
295 			'-action': 'xf_schemabrowser_preview_row',
296 			'-cursor': sb.query['-cursor'] || 0
297 		};
298 		$.get(url, q, function(res){
299 			try {
300 				if ( res.code == 200 ){
301 					var values = res.values;
302 					sb.previewValues = values;
303 					$('li[xf-schemabrowser-macro]', el).each(function(){
304 						var macro = $(this).attr('xf-schemabrowser-macro');
305 						if ( typeof(values[macro]) != 'undefined' ){
306 							$('span.xf-preview', this).html(createPreview(values[macro]));
307 						} else {
308 							$('span.xf-preview', this).text('');
309 						}
310 					});
311 					sb.statusButton.setLabel("Now Showing Record #"+(parseInt(sb.query['-cursor']||0)+1));
312 				} else if ( res.message ){
313 					throw res.message;
314 				} else {
315 					throw 'Failed to load preview.  See server error log for details.';
316 				}
317 			} catch (e){
318 				alert(e);
319 			}
320 		});
321 	}

323 	/**
324 	 * Utility function to shorten a value so that it can be displayed in the limited
325 	 * preview space in the tree (next to its associated field node).  It strips out
326 	 * all HTML tags and trims it to a maximum 20 characters.
327 	 *
328 	 *
329 	 * @param {String} val The field value to be shortened.
330 	 * @return {String} The shortened field value.
331 	 */
332 	function createPreview(val){
333 		if ( val == null ) return '';
334 		val = (''+val).replace(/<[^>]+>/, '');
335 		var sval = val.substring(0,20);
336 		if ( sval.length < val.length ) sval +='...';
337 		return sval;
338 	}
