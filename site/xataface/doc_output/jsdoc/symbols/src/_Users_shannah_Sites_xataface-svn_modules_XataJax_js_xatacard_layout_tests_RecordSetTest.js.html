 64 		function testRecordGettersAndSetters(){
 65 			
 66 			var r = new Record();
 67 			var s = new Schema();
 68 			
 69 			var schemaChangeEvent = null;
 70 			
 71 			r.addPropertyChangeListener(new PropertyChangeListener({
 72 				propertyChange: function(evt){
 73 					schemaChangeEvent = evt;
 74 				}
 75 			}));
 76 			
 77 			r.setSchema(s);
 78 			
 79 			self.assertTrue('record propertyChangeEvent(schema)', XataJax.instanceOf(schemaChangeEvent, PropertyChangeEvent));
 80 			self.assertEquals('record propertyChangeEvent(schema) propertyName', schemaChangeEvent.propertyName, 'schema');
 81 			self.assertEquals('record propertyChangeEvent(schema) oldValue', schemaChangeEvent.oldValue, null);
 82 			self.assertEquals('record propertyChangeEvent(schema) newValue', schemaChangeEvent.newValue, s);
 83 			self.assertEquals('record propertyChangeEvent(schema) source', schemaChangeEvent.source, r);
 84 			
 85 			var code = null;
 86 			try {
 87 				r.setValue('foo','bar');
 88 			} catch (ex){
 89 				code = ex.getCode();
 90 			}
 91 			self.assertEquals('record set value not in schema', code, XataJax.errorcodes.PATH_NOT_IN_SCHEMA);
 92 			
 93 			code = null;
 94 			try {
 95 				r.getValue('foo');
 96 			} catch (ex){
 97 				code = ex.getCode();
 98 			}
 99 			self.assertEquals('record get value not in schema', code, XataJax.errorcodes.PATH_NOT_IN_SCHEMA);
100 			
101 			
102 			var nameField = new Field({path:'name'});
103 			s.addField(nameField);
104 			self.assertTrue('record isDirty before changes', !r.isDirty());
105 			self.assertTrue('record valueChanged before changes', !r.valueChanged('name'));
106 			r.setValue('name','Steve');
107 			self.assertEquals('record set/get value in schema', r.getValue('name'), 'Steve');
108 			self.assertTrue('record isDirty after changes', r.isDirty());
109 			self.assertTrue('record valueChanged after changes', r.valueChanged('name'));
110 			
111 			var snapshot = r.getSnapshot();
112 			self.assertEquals('record snapshot name value', snapshot.name, null);
113 			
114 			// test out the triggers.
115 			
116 			var beforeSetValueEvent = null;
117 			var afterSetValueEvent = null;
118 			var afterSchemaSetValueEvent = null;
119 			var beforeSchemaSetValueEvent = null;
120 			
121 			r.bind('beforeSetValue', function(evt){
122 				//alert(evt.newValue);
123 				beforeSetValueEvent = evt;
124 			});
125 			
126 			r.bind('afterSetValue', function(evt){
127 				afterSetValueEvent = evt;
128 			});
129 			
130 			s.bind('afterSetValue', function(evt){
131 				afterSchemaSetValueEvent = evt;
132 			});
133 			
134 			s.bind('beforeSetValue', function(evt){
135 				beforeSchemaSetValueEvent = evt;
136 			});
137 			
138 			
139 			
140 			r.setValue('name', 'Steve'); // shouldn't fire a trigger - no value change
141 			self.assertEquals('record beforeSetValue event not fired', beforeSetValueEvent, null);
142 			
143 			
144 			
145 			r.setValue('name', 'Dave'); // should fire trigger
146 			//alert(beforeSetValueEvent.newValue);
147 			//alert(XataJax.instanceOf(beforeSetValueEvent, PropertyChangeEvent));
148 			//alert(PropertyChangeEvent);
149 			//alert(beforeSetValueEvent.constructor);
150 			self.assertTrue('record beforeSetValue event fired', XataJax.instanceOf(beforeSetValueEvent, PropertyChangeEvent));
151 			self.assertEquals('record beforeSetValue event property name', beforeSetValueEvent.propertyName, 'name');
152 			self.assertEquals('record beforeSetValue event poperty old value', beforeSetValueEvent.oldValue, 'Steve');
153 			self.assertEquals('record beforeSetValue event property new value', beforeSetValueEvent.newValue, 'Dave');
154 			self.assertEquals('record beforeSetValue event source', beforeSetValueEvent.source, r);
155 			self.assertEquals('record getValue after setting value', r.getValue('name'), 'Dave');
156 		
157 		
158 			self.assertTrue('record afterSetValue event fired', XataJax.instanceOf(afterSetValueEvent, PropertyChangeEvent));
159 			self.assertEquals('record afterSetValue event property name', afterSetValueEvent.propertyName, 'name');
160 			self.assertEquals('record afterSetValue event poperty old value', afterSetValueEvent.oldValue, 'Steve');
161 			self.assertEquals('record afterSetValue event property new value', afterSetValueEvent.newValue, 'Dave');
162 			self.assertEquals('record afterSetValue event source', afterSetValueEvent.source, r);
163 			
164 		
165 			self.assertTrue('record afterSetValue schema event fired', XataJax.instanceOf(afterSchemaSetValueEvent, PropertyChangeEvent));
166 			self.assertEquals('record afterSchemaSetValue event property name', afterSchemaSetValueEvent.propertyName, 'name');
167 			self.assertEquals('record afterSchemaSetValue event poperty old value', afterSchemaSetValueEvent.oldValue, 'Steve');
168 			self.assertEquals('record afterSchemaSetValue event property new value', afterSchemaSetValueEvent.newValue, 'Dave');
169 			self.assertEquals('record afterSchemaSetValue event source', afterSchemaSetValueEvent.source, r);
170 			
171 			self.assertTrue('record beforeSetValue schema event fired', XataJax.instanceOf(afterSchemaSetValueEvent, PropertyChangeEvent));
172 			self.assertEquals('record beforeSchemaSetValue event property name', beforeSchemaSetValueEvent.propertyName, 'name');
173 			self.assertEquals('record beforeSchemaSetValue event poperty old value', beforeSchemaSetValueEvent.oldValue, 'Steve');
174 			self.assertEquals('record beforeSchemaSetValue event property new value', beforeSchemaSetValueEvent.newValue, 'Dave');
175 			self.assertEquals('record beforeSchemaSetValue event source', beforeSchemaSetValueEvent.source, r);
176 			
177 			
178 			// Try introducing a separate schema.
179 			var s2 = new Schema();
180 			var nameField2 = new Field({path: 'name'});
181 			s2.addField(nameField2);
182 			
183 			r.setSchema(s2);
184 			
185 			self.assertTrue('record propertyChangeEvent(schema) second time', XataJax.instanceOf(schemaChangeEvent, PropertyChangeEvent));
186 			self.assertEquals('record propertyChangeEvent(schema) second time propertyName', schemaChangeEvent.propertyName, 'schema');
187 			self.assertEquals('record propertyChangeEvent(schema) second time oldValue', schemaChangeEvent.oldValue, s);
188 			self.assertEquals('record propertyChangeEvent(schema) second time newValue', schemaChangeEvent.newValue, s2);
189 			self.assertEquals('record propertyChangeEvent(schema) second time source', schemaChangeEvent.source, r);
190 			
191 			
192 			beforeSchemaSetValueEvent = null;
193 			r.setValue('name', 'Daniel');
194 			
195 			// We are checking here to make sure that the beforeSetValue event was properly
196 			// unbound from the old schema when the schema was changed.
197 			self.assertEquals('record beforeSetValueEvent in schema after changing to different schema', beforeSchemaSetValueEvent, null);			
198 			
199 			
200 			// Now let's do some datasource testing.
201 			var dsException = null;
202 			try {
203 				r.commit();
204 			} catch (ex){
205 				dsException = ex;
206 			}
207 			
208 			//alert(dsException);
209 			//alert(dsException.constructor);
210 			self.assertTrue('record committing without datasource exception', XataJax.instanceOf(dsException, Exception));
211 			self.assertEquals('record committing without datasource exception code', dsException.getCode(), XataJax.errorcodes.NOT_BOUND_TO_DATASOURCE);

218 		function testSchema(){
219 			var s = new Schema();
220 			
221 			self.assertEquals('schema createRecord', s.createRecord().constructor, Record);
222 			
223 			var nameField = new Field({path:'name'});
224 			var addressField = new Field({path:'address'});
225 			var phoneField = new Field({path:'phone'});
226 			
227 			s.addField(nameField);
228 			self.assertEquals('schema getField("name")', s.getField('name'), nameField);
229 			
230 			nameField.setPath('name2');
231 			self.assertEquals('schema getField("name2")', s.getField('name2'), nameField);
232 			self.assertEquals('schema getField("name") after changing name', s.getField('name'), null);
233 			
234 			self.assertEquals('field property change listeners before removal', nameField.getPropertyChangeListeners().length, 1);
235 			
236 			s.removeField(nameField);
237 			self.assertEquals('schema getField("name2") after removal', s.getField('name2'), null);
238 			
239 			// Make sure that the listener has been removed.
240 			self.assertEquals('field property change listeners after removal', nameField.getPropertyChangeListeners().length, 0);
241 			
242 			
243 			
244 			
245

249 		function testField(){
250 		
251 			var fld = new Field({
252 				path: 'path/to/me'
253 			});
254 			
255 			self.assertEquals('field getPath after constructor', fld.getPath(), 'path/to/me');
256 			fld.setPath('foo');
257 			self.assertEquals('field getPath after setPath', fld.getPath(), 'foo');
258 			
259 			
260 			// Now to test property change events
261 			var numPropertyChangeCalls = 0;
262 			var listener = new PropertyChangeListener({
263 				propertyChange: function(evt){
264 					var name = evt.propertyName;
265 					var source = evt.source;
266 					var old = evt.oldValue;
267 					var newVal = evt.newValue;
268 					self.assertEquals('field property change for path name', name, 'path');
269 					self.assertEquals('field property change old value bar', old, 'foo');
270 					self.assertEquals('field property change new value var', newVal, 'bar');
271 					self.assertEquals('field property change source', source, fld);
272 					numPropertyChangeCalls++;
273 				}
274 			});
275 			
276 			fld.addPropertyChangeListener(listener);
277 			fld.setPath('bar');
278 			self.assertEquals('field property change number of propertyChangeCalls', numPropertyChangeCalls, 1);
279 		}
