  7 	 * Import globals to local scope.
  8 	 */

 10 	var UndoableEdit = XataJax.undo.UndoableEdit;
 11 	var UndoableEditListener = XataJax.undo.UndoableEditListener;
 12 	var UndoableEditEvent = XataJax.undo.UndoableEditEvent;
 13 	var Exception = XataJax.Exception;
 14 	
 15 	
 16 	/**
 17 	 * Register the public API
 18 	 */
 19 	XataJax.undo.UndoManager = UndoManager;
 20 	
 21 	
 22 	/**
 23 	 * Error code for exceptions thrown when there are no edits left
 24 	 * to redo, but the method call implies that we expect there to be.
 25 	 * e.g. the redoTo(UndoableEdit) method.
 26 	 */
 27 	XataJax.errorcodes.REDO_STACK_EMPTY = XataJax.nextErrorCode();
 28 	
 29 	/**
 30 	 * Error code for exceptions thrown when there are no edits left to
 31 	 * undo, but the method call implies that we expect there to be.
 32 	 * e.g. undoTo(UndoableEdit) method.
 33 	 */
 34 	XataJax.errorcodes.UNDO_STACK_EMPTY = XataJax.nextErrorCode();
 35 	
 36 	
 37 	/**
 38 	 * The UndoManager constructor.  This is essentially a port of the Java
 39 	 * Swing UndoManager class into javascript so that we can add Undo/Redo
 40 	 * support to a javascript application.
 41 	 *
 42 	 * @see http://download.oracle.com/javase/6/docs/api/javax/swing/undo/UndoManager.html
 43 	 * for more details about the Java API that we loosely copied here.
 44 	 *
 45 	 * @constructor
 46 	 """
 47 	 new UndoManager();
 48 	 """
 49 	 */
 50 	function UndoManager(o){
 51 		
 52 		/**
 53 		 * Extends the UndoableEdit class.
 54 		 */
 55 		XataJax.extend(this, new UndoableEditListener(o));
 56 		XataJax.extend(this, new UndoableEdit(o));
 57 		
 58 		var publicAPI = {
 59 		
 60 			setLimit: setLimit,
 61 			getLimit: getLimit,
 62 			trimForLimit: trimForLimit,
 63 			addEdit: addEdit,
 64 			canRedo: canRedo,
 65 			canUndo: canUndo,
 66 			canUndoOrRedo: canUndoOrRedo,
 67 			discardAllEdits: discardAllEdits,
 68 			editToBeRedone: editToBeRedone,
 69 			editToBeUndone: editToBeUndone,
 70 			redoTo: redoTo,
 71 			undoOrRedo: undoOrRedo,
 72 			undoInsignificant: undoInsignificant,
 73 			undo: undo,
 74 			redo: redo,
 75 			undoTo: undoTo,
 76 			getUndoPresentationName: getUndoPresentationName,
 77 			getRedoPresentationName: getRedoPresentationName,
 78 			getUndoOrRedoPresentationName: getUndoOrRedoPresentationName,
 79 			end: end
 80 		};
 81 		
 82 		/**
 83 		 * Register public methods.
 84 		 */
 85 		XataJax.publicAPI(this, publicAPI);
 86 		
 87 		
 88 		/**
 89 		 * Private member variables.-------------------------------------------------
 90 		 */
 91 		 
 92 		/**
 93 		 * @type array(UndoableEdit) Stack of edits that can be undone.
 94 		 */
 95 		var undoStack = [];
 96 		
 97 		/**
 98 		 * @type array(UndoableEdit) Stack of edits that can be redone.
 99 		 */
100 		var redoStack = [];
101 		
102 		/**
103 		 * @type boolean A flag to indicate if this manager has been ended and
104 		 *		ceases to act as an undoManager anymore.  If this flag is set
105 		 * 		then this will function like a regular UndoableEdit object.
106 		 */
107 		var ended = false;
108 		
109 		/**
110 		 * @type int
111 		 * The maxiumm number of edits that can be stored.
112 		 */
113 		var limit = 30;
114 		
115 		
116 		/**
117 		 * END PRIVATE MEMBERS-------------------------------------------------------
118 		 */
119 		 
120 		/**
121 		 * Ends this manager so that it will be treated just like a normal edit.
122 		 */
123 		function end(){
124 			ended = true;
125 		}
126 		
127 		
128 		/**
129 		 * Sets the maximum number of edits that can be undone.
130 		 * @param {int} l
131 		 */
132 		function setLimit(l){
133 			limit = l;
134 			this.trimForLimit();
135 		}
136 		
137 		
138 		/**
139 		 * Gets the maximum number of edits that can be undone.
140 		 *
141 		 * @returns {int}
142 		 */
143 		function getLimit(){
144 			return limit;
145 		}
146 		
147 		
148 		/**
149 		 * Trims the undo stack down to only contain the maximum limit of 
150 		 * edits (@see getLimit()).  Edits are deleted from oldest to newest
151 		 * and their die() method is called as they are deleted.
152 		 */
153 		function trimForLimit(){
154 			while ( undoStack.length > this.getLimit() ){
155 				var edit = undoStack.shift();
156 				edit.die();
157 			}
158 		}
159 		
160 		/**
161 		 * Adds an edit to the undo manager.
162 		 * @param {UndoableEdit} The UndoableEdit object that is being added. This 
163 		 *		method also has the effect of clearing the redoStack.
164 		 */
165 		function addEdit(edit){
166 			// we need to clear the redoStack out
167 			redoStack = [];
168 			undoStack.push(edit);
169 			this.trimForLimit();
170 		}
171 		
172 		
173 		/**
174 		 * Returns true if edits may be redone. If end has been invoked, this returns 
175 		 *	the value from super. Otherwise, this returns true if there are any 
176 		 * edits to be redone (editToBeRedone returns non-null).
177 		 * 
178 		 * @returns {boolean}
179 		 */
180 		function canRedo(){
181 			if ( ended ){
182 				return this.getSuper(UndoableEdit).canRedo();
183 			}
184 			if ( redoStack.length == 0 ){
185 				return false;
186 			}
187 			for ( var i=redoStack.length-1; i>=0; i--){
188 				if ( !redoStack[i].canRedo() ){
189 					return false;
190 				}
191 				if ( redoStack[i].isSignificant() ){
192 					return true;
193 				}
194 			}
195 			return false;
196 		}
197 		
198 		/**
199 		 * Returns true if edits may be undone. If end has been invoked, this returns 
200 		 * the value from super. Otherwise this returns true if there are any edits 
201 		 * to be undone (editToBeUndone returns non-null).
202 		 *
203 		 * @returns {boolean}
204 		 */
205 		function canUndo(){
206 			if ( ended ){
207 				return this.getSuper(UndoableEdit).canUndo();
208 			}
209 		
210 			if ( undoStack.length == 0 ){
211 				return false;
212 			}
213 			for ( var i=undoStack.length-1; i>=0; i-- ){
214 				if ( !undoStack[i].canUndo() ){
215 					return false;
216 				}
217 				if ( undoStack[i].isSignificant() ){
218 					return true;
219 				}
220 			}
221 			return false;
222 		}
223 		
224 		/**
225 		 * Returns true if it is possible to invoke undo or redo.
226 		 *
227 		 * @returns {boolean}
228 		 */
229 		function canUndoOrRedo(){
230 			return (this.canUndo()||this.canRedo());
231 		}
232 		
233 		
234 		/**
235 		 * Empties the undo manager sending each edit a die message in the process.
236 		 */
237 		function discardAllEdits(){
238 			while ( undoStack.length > 0 ){
239 				undoStack.shift().die();
240 			}
241 			while ( redoStack.length > 0 ){
242 				redoStack.shift().die();
243 			}
244 		}
245 		
246 		/**
247 		 * Returns the the next significant edit to be redone if redo is invoked. 
248 		 * This returns null if there are no edits to be redone.
249 		 *
250 		 * @returns {UndoableEdit}
251 		 */
252 		function editToBeRedone(){
253 			if ( !this.canRedo() ) return null;
254 			for ( var i=redoStack.length-1; i>=0; i-- ){
255 				if ( redoStack[i].canRedo() && redoStack[i].isSignificant() ){
256 					return redoStack[i];
257 				}
258 			}
259 			return null;
260 		}
261 		
262 		
263 		/**
264 		 * Returns the the next significant edit to be undone if undo is invoked. 
265 		 * This returns null if there are no edits to be undone.
266 		 *
267 		 * @returns {UndoableEdit}
268 		 */
269 		function editToBeUndone(){
270 			if ( !this.canUndo() ) return null;
271 			for ( var i=undoStack.length-1; i>=0; i-- ){
272 				if ( undoStack[i].canUndo() && undoStack[i].isSignificant() ){
273 					return undoStack[i];
274 				}
275 			}
276 			
277 			return null;
278 		}
279 		
280 		
281 		/**
282 		 * Redoes all changes from the index of the next edit to edit.
283 		 *
284 		 * @param {UndoableEdit} edit The edit up to which is to be redone (inclusive).
285 		 * @throws Exception(XataJax.errorcodes.CANNOT_REDO_EXCEPTION)
286 		 */
287 		function redoTo(edit){
288 			if ( redoStack.length == 0 ){
289 				throw new Exception({
290 					message: 'No edits available to redo.',
291 					code: XataJax.errorcodes.CANNOT_REDO_EXCEPTION
292 				});
293 			}
294 			var curr = null;
295 			while ( redoStack.length > 0 ){
296 				curr = redoStack.pop();
297 				
298 				curr.redo();
299 				undoStack.push(curr);
300 				if ( curr == edit ){
301 					return;
302 				}
303 			}
304 			
305 
306 			throw new Exception({
307 				message: 'Redo stack is empty',
308 				code: XataJax.errorcodes.REDO_STACK_EMPTY
309 			});
310 
311 		}
312 		
313 		/**
314 		 * Convenience method that invokes one of undo or redo. If any edits have been 
315 		 * undone this invokes redo, otherwise it invokes undo.
316 		 *
317 		 * @returns {boolean}
318 		 */
319 		function undoOrRedo(){
320 			if ( this.canRedo() ){
321 				this.redo();
322 			} else if ( this.canUndo() ){
323 				this.undo();
324 			} else {
325 				throw new Exception({
326 					message: 'No undos or redos to do',
327 					code: XataJax.errorcodes.REDO_STACK_EMPTY
328 				});
329 			}
330 		}
331 		
332 		/**
333 		 * Undoes all of the insignificant edits at the top of the stack.
334 		 * After running this method either the undo stack is empty or
335 		 * the very top item is a significant edit.
336 		 *
337 		 * @returns {void}
338 		 */
339 		function undoInsignificant(){
340 			var curr = null;
341 			while ( undoStack.length > 0 ){
342 				curr = undoStack.pop();
343 				if ( curr.isSignificant() ){
344 					undoStack.push(curr);
345 					return;
346 				}
347 				curr.undo();
348 				redoStack.push(curr);
349 				
350 			}
351 			
352 		}
353 		
354 		
355 		/**
356 		 * Undoes the appropriate edits. If end has been invoked this calls through 
357 		 * to the superclass, otherwise this invokes undo on all edits between the 
358 		 * index of the next edit and the last significant edit, updating the index of 
359 		 * the next edit appropriately.
360 		 *
361 		 * @throws {Exception(XataJax.errorcodes.CANNOT_UNDO_EXCEPTION)}
362 		 */
363 		function undo(){
364 			if ( ended ){
365 				return this.getSuper(UndoableEdit).undo();
366 			}
367 			var nextEdit = this.editToBeUndone();
368 			if ( nextEdit != null ){
369 				this.undoTo(nextEdit);
370 				this.undoInsignificant();
371 			}
372 		}
373 		
374 		
375 		/**
376 		 * Redoes the appropriate edits. If end has been invoked this calls through 
377 		 * to the superclass. Otherwise this invokes redo on all edits between the 
378 		 * index of the next edit and the next significant edit, updating the index of 
379 		 * the next edit appropriately.
380 		 *
381 		 * @throws {Exception(XataJax.errorcodes.CANNOT_UNDO_EXCEPTION)}
382 		 */
383 		function redo(){
384 			if ( ended ){
385 				return this.getSuper(UndoableEdit).redo();
386 			}
387 			var nextEdit = this.editToBeRedone();
388 			this.redoTo(nextEdit);
389 			this.redoInsignificant();
390 		}
391 		
392 		/**
393 		 * Undoes all edits up to and including the specified edit.
394 		 *
395 		 * @param {UndoableEdit} edit
396 		 * @throws {Exception(XataJax.errorcodes.CANNOT_UNDO_EXCEPTION)}
397 		 */
398 		function undoTo(edit){
399 			var curr = null;
400 			while ( undoStack.length > 0 ){
401 				curr = undoStack.pop();
402 				curr.undo();
403 				redoStack.push(curr);
404 				if ( curr == edit ){
405 					return;
406 				}
407 			}
408 			
409 			throw new Exception({
410 				message: 'Undo stack is empty and we still haven\'t reached the edit we want to remove.',
411 				code: XataJax.errorcodes.UNDO_STACK_EMPTY = XataJax.nextErrorCode()
412 			});
413 		}
414 		
415 		
416 		/**
417 		 * Returns a description of the undoable form of this edit. If end has been invoked 
418 		 * this calls into super. Otherwise if there are edits to be undone, this returns 
419 		 * the value from the next significant edit that will be undone.
420 		 *
421 		 * @returns {String}
422 		 */
423 		function getUndoPresentationName(){
424 			if ( ended ){
425 				return this.getSuper(UndoableEdit).getUndoPresentationName();
426 			}
427 			var next = this.editToBeUndone();
428 			if ( next != null ){
429 				return next.getUndoPresentationName();
430 			} else {
431 				return this.getSuper(UndoableEdit).getUndoPresentationName();
432 			}
433 			
434 		}
435 		
436 		/**
437 		 * Returns a description of the redoable form of this edit. If end has been 
438 		 * invoked this calls into super. Otherwise if there are edits to be redone, 
439 		 * this returns the value from the next significant edit that will be redone. 
440 		 * 
441 		 * @returns {String}
442 		 */
443 		function getRedoPresentationName(){
444 			if ( ended ){
445 				return this.getSuper(UndoableEdit).getRedoPresentationName();
446 			}
447 			var next = this.editToBeRedone();
448 			if ( next != null ){
449 				return next.getRedoPresentationName();
450 			} else {
451 				return this.getSuper(UndoableEdit).getUndoPresentationName();
452 			}
453 		}
454 		
455 		/**
456 		 * Convenience method that returns either getUndoPresentationName or 
457 		 * getRedoPresentationName. If the index of the next edit equals the size 
458 		 * of the edits list, getUndoPresentationName is returned, otherwise 
459 		 * getRedoPresentationName is returned.
460 		 *
461 		 * @returns {String}
462 		 */
463 		 
464 		function getUndoOrRedoPresentationName(){
465 			if ( this.canRedo() ){
466 				return this.getRedoPresentationName();
467 			} else if ( this.canUndo() ){
468 				return this.getUndoPresentationName();
469 			} else {
470 				return null;
471 			}
472 		}

475 		 * UndoableEditListener method
476 		 *
477 		 * @param {UndoableEditEvent} event
478 		 */
479 		function undoableEditHappened(event){
480 			this.addEdit(event.edit);
481 		}
