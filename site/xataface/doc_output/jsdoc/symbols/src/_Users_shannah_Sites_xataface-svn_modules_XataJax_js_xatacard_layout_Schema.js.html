112 			propertyChange: function(evt){
113 				// Perform stuff in here to handle when fields are

115 				// when the field's path is changed so we can update

119 					//alert("Property change: "+evt.oldValue+' -> '+evt.newValue);

121 					delete fields[evt.oldValue];

134 					var schema = evt.source;
135 					delete subSchemas[evt.oldValue];
136 					subSchemas[evt.newValue] = schema;
137 				}
138 			}

279 		 * Decorates a record of this schema.  This is generally called on to decorate
280 		 * records created with createRecord, but you could use this on any record to
281 		 * essentially cast the record into this schema.
282 		 *

287 			var self = this;
288 			//record.setSchema(this);
289 			record.addPropertyChangeListener(recordListener);
290 			//alert('here');
291 			record.bind('validate.Schema', function(e){self.validate(e);});
292 			record.bind('beforeSave.Schema', function(e){self.beforeSave(e);});
293 			record.bind('afterSave.Schema', function(e){self.afterSave(e);});
294 			record.bind('beforeInsert.Schema', function(e){self.beforeInsert(e);});
295 			record.bind('afterInsert.Schema', function(e){self.afterInsert(e);});
296 			record.bind('beforeUpdate.Schema', function(e){self.beforeUpdate(e);});
297 			record.bind('afterUpdate.Schema', function(e){self.afterUpdate(e);});
298 			
299 			record.bind('beforeDelete.Schema', function(e){self.beforeDelete(e);});
300 			record.bind('afterDelete.Schema', function(e){self.afterDelete(e);});
301 			record.bind('beforeSetValue.Schema', function(e){self.beforeSetValue(e);});
302 			record.bind('afterSetValue.Schema', function(e){self.afterSetValue(e);});
303 			this.trigger('decorateRecord', new RecordEvent({
304 				record: record,

338 		 * Adds a field to this schema.
339 		 * @param {Field} fld The field to be added.
340 		 * @returns {Schema} Self for chaining.
341 		 */
342 		function addField(fld){
343 			if ( typeof(fields[fld.getPath()]) != 'undefined' ){
344 				var old = fields[fld.getPath()];
345 				if ( old == fld ) return this;
346 				else {
347 					this.removeField(old);
348 				}
349 			}
350 			fields[fld.getPath()] = fld;
351 			
352 			fld.addPropertyChangeListener(fieldListener);
353 			this.trigger('fieldAdded', new SchemaFieldEvent({
354 				schema: this,
355 				field: fld,
356 				action: 'add'
357 			}));
358 			return this;
359 		}

363 		 * @param {Field} fld The field to be removed.
364 		 * @returns {Schema} Self for chaining.
365 		 */

367 			delete fields[fld.getPath()];
368 			fld.removePropertyChangeListener(fieldListener);
369 			this.trigger('fieldRemoved', new SchemaFieldEvent({
370 				schema: this,
371 				field: fld,
372 				action: 'remove'
373 			}));
374 			return this;

377 		/**
378 		 * Gets the field of this schema located at the specified path.
379 		 * @param {String} path The path to the field.
380 		 * @param {boolean} Whether to search subschemas.
381 		 */
382 		 function getField(path, searchSubschemas){
383 			if ( typeof(searchSubschemas) == 'undefined' ){
384 				searchSubschemas = false;
385 			}
386 			if ( typeof(fields[path]) != 'undefined' ){
387 				return fields[path];
388 			}
389 			if ( searchSubschemas ){
390 				for ( var p in subSchemas ){
391 					if ( path.indexOf(p) == 0 ){
392 						var field = subSchemas[p].getField(path.substr(p.length));
393 						if ( field != null ){
394 							return field;
395 						}
396 					}
397 				}
398 			}
399 			return null;
400 		}

415 		 * @return {Schema} Self for chaining.

417 		 */
418 		function addSubSchema(schema){
419 			if ( typeof(subSchemas[schema.getPath()]) != 'undefined' ){
420 				var old = subSchemas[schema.getPath()];
421 				if ( old == schema ) return this;
422 				else this.removeSubSchema(old);
423 			}
424 			subSchemas[schema.getPath()] = schema;
425 			schema.addPropertyChangeListener(subSchemaListener);
426 			this.trigger('subSchemaAdded', new SchemaSubSchemaEvent({
427 				schema: this,
428 				subSchema: schema,
429 				action: 'add'
430 			}));
431 			return this;
432 		}

453 		 * Returns the subschema at the specified path.
454 		 * @param {String} path The path to the subschema.
455 		 */
456 		function getSubSchema(path, searchSubschemas){
457 			if ( typeof(searchSubschemas) == 'undefined' ){
458 				searchSubschemas = false;
459 			}
460 			if ( typeof(subSchemas[path]) != 'undefined' ){
461 				return subSchemas[path];
462 			}
463 			if ( searchSubschemas ){
464 				for ( var p in subSchemas ){
465 					if ( path.indexOf(p) == 0 ){
466 						var out = subSchemas[p].getSubSchema(path.subtr(p.length));
467 						if ( out != null ){
468 							return out;
469 						}
470 					}
471 				}
472 			}
473 			return null;
474 		}
