 27 		function assertTrue(testname, expr){

 29 				passed.push(testname);
 30 			} else {

 32 			}

 37 		}

 39 		function assertEquals(testname, expected, actual){

 41 				passed.push(testname);
 42 			} else {
 43 				failed.push(testname + ' failed [Expected '+expected+' ; Actual '+actual+']');
 44 			}
 45 		}
 46 	
 47 		function runTests(outputDiv){
 48 			//alert(this.getTests);
 49 			var tests = this.getTests();

 51 				this();
 52 			});
 53

 55 			$.each(passed, function(){

 58 				$(row).css('color', 'white');
 59 			
 60 				//$(row).text(this);
 61 				row.appendChild(document.createTextNode(this+ ' Passed'));

 63 			});

 67 				$(row).css('background-color', 'red');
