1 //require <xatajax.ui.tk/LayoutManager.js>
  2 //require <xatajax.ui.tk/ComponentEvent.js>
  3 //require <xatajax.ui.tk/ComponentListener.js>
  4 //require-css <xatajax.ui.tk/layout/ColumnLayout.css>
  5 (function(){
  6 	var $ = jQuery;
  7 	var LayoutManager = XataJax.ui.tk.LayoutManager;
  8 	var ComponentListener = XataJax.ui.tk.ComponentListener;
  9 	var ComponentEvent = XataJax.ui.tk.ComponentEvent;
 10 	var Component = XataJax.ui.tk.Component;
 11 	
 12 	XataJax.ui.tk.layout.ColumnLayout = ColumnLayout;
 13 	
 14 	
 15 	/**
 16 	 * @constructor
 17 	 */
 18 	function ColumnLayout(o){
 19 		
 20 		
 21 		XataJax.extend(this, new LayoutManager(o));
 22 		
 23 		var publicAPI = {
 24 			install: install,
 25 			uninstall: uninstall,
 26 			update: update,
 27 			childAdded: childAdded,
 28 			childRemoved: childRemoved
 29 			
 30 		};
 31 		
 32 		XataJax.publicAPI(this, publicAPI );
 33 		
 34 		
 35 		/**
 36 		 * @param {Component} c
 37 		 */
 38 		function install(c){
 39 			this.getSuper(LayoutManager).install(c);
 40 			
 41 			$(c.getElement()).addClass('xatajax-ui-layout-columnlayout');
 42 			this.getComponent().update();
 43 		}
 44 		
 45 		/**
 46 		 * @param {Component} c
 47 		 */
 48 		function uninstall(c){
 49 			this.getSuper(LayoutManager).uninstall(c);
 50 			$(c.getElement()).removeClass('xatajax-ui-layout-columnlayout');
 51 		}
 52 		
 53 		/**
 54 		 * Refreshes the layout.
 55 		 */
 56 		function update(){
 57 			
 58 			if ( this.getComponent() == null ) return;
 59 			
 60 			this.getSuper(LayoutManager).update();
 61 			
 62 			$(this.getComponent().getElement()).children().detach();
 63 			
 64 			var table = document.createElement('table');
 65 			
 66 			var tr = document.createElement('tr');
 67 			table.appendChild(tr);
 68 			$(table).addClass('xatajax-ui-layout-columnlayout-table');
 69 			var i=1;
 70 			$.each(this.getComponent().getChildComponents(), function(){
 71 				
 72 				var td = document.createElement('td');
 73 				$(td)
 74 					.addClass('xatajax-ui-component-slot')
 75 					.addClass('xatajax-ui-layout-columnlayout-table-column-'+(i++))
 76 					.data('xatajax-component-slot', i-1)
 77 					;
 78 				tr.appendChild(td);
 79 				td.appendChild(this.getElement());
 80 			});
 81 			this.getComponent().getElement().appendChild(table);
 82 			
 83 		}
 84 		
 85 		
 86 		/**
 87 		 * @interface ComponentListener
 88 		 * @param {ComponentEvent} event
 89 		 */
 90 		function childAdded(event){
 91 			
 92 			this.getComponent().update();
 93 		}
 94 		
 95 		
 96 		/**
 97 		 * @interface ComponentListener
 98 		 * @param {ComponentEvent} event
 99 		 */
100 		function childRemoved(event){
101 			this.getComponent().update();
102 		}
103 		
104 	
105 	}
106 })();
