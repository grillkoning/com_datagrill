  2 //require <xatajax.ui.tk/Button.js>
  3 //require <xatajax.ui.tk/layout/FlowLayout.js>
  4 //require-css <jquery-ui/jquery-ui.css>
  5 //require-css <xatajax.ui.tk/ButtonGroup.css>
  6 
  7 (function(){
  8 	
  9 	
 10 	var $ = jQuery;
 11 	var Component = XataJax.ui.tk.Component;
 12 	var Button = XataJax.ui.tk.Button;
 13 	var FlowLayout = XataJax.ui.tk.layout.FlowLayout;
 14 	
 15 	
 16 	XataJax.ui.tk.ButtonGroup = ButtonGroup;
 17 	
 18 	
 19 	/**
 20 	 * A ButtonGroup component that is handy building a toolbar.  The concept 
 21 	 * for this button group is borrowed from Filament Group:
 22 	 * <a href="http://www.filamentgroup.com/lab/styling_buttons_and_toolbars_with_the_jquery_ui_css_framework/">Here</a>.
 23 	 *
 24 	 * @constructor
 25 	 */
 26 	function ButtonGroup(o){
 27 	
 28 	
 29 		/**
 30 		 * A setting to decide whether this button group allows only a single select.
 31 		 *
 32 		 * @type {boolean}
 33 		 */
 34 		var singleSelect = true;
 35 	
 36 		XataJax.extend(this, new Component(o));
 37 		XataJax.publicAPI(this, {
 38 			setSingleSelect: setSingleSelect,
 39 			isSingleSelect: isSingleSelect,
 40 			decorateElement: decorateElement,
 41 			add: add,
 42 			remove: remove
 43 		});
 44 		
 45 		this.setLayout(new FlowLayout());
 46 		
 47 		
 48 		/**
 49 		 * Sets whether this group allows only single selection.
 50 		 *
 51 		 * @param {boolean} s
 52 		 * @returns {ButtonGroup} Self for chaining.
 53 		 */
 54 		function setSingleSelect(s){
 55 		
 56 			if ( s != singleSelect ){
 57 				var old = singleSelect;
 58 				singleSelect = s;
 59 				if ( s ){
 60 					$(e)
 61 						.removeClass('xatajax-buttonset-multi')
 62 						.addClass('xatajax-buttonset-single');
 63 				} else {
 64 					$(e)
 65 						.removeClass('xatajax-buttonset-single')
 66 						.addClass('xatajax-buttonset-multi');
 67 				}
 68 				this.firePropertyChange('singleSelect', old, s);
 69 			}
 70 			return this;
 71 		}
 72 		
 73 		/**
 74 		 * Checks if this group allows only single selection.
 75 		 * @returns {boolean}
 76 		 */
 77 		function isSingleSelect(){
 78 			return singleSelect;
 79 		}
 80 		
 81 		/**
 82 		 * Decorates the buttongroup's element.
 83 		 * @override
 84 		 * @param {HTMLElement} e The element for this component.
 85 		 * @returns {ButtonGroup} Self for chaining.
 86 		 */
 87 		function decorateElement(e){
 88 			this.getSuper(Component).decorateElement(e);
 89 			$(e).addClass('xatajax-buttonset');
 90 				
 91 			if ( singleSelect ){
 92 				$(e).addClass('xatajax-buttonset-single');
 93 			} else {
 94 				$(e).addClass('xatajax-buttonset-multi');
 95 			}
 96 			
 97 			return this;
 98 		}
 99 		
100 		
101 		function buttonMouseDown(){
102 			if ( !$(this).hasClass('ui-state-disabled') ){
103 				$(this)
104 					.parents('.xatajax-buttonset-single').first()
105 						.find(".xatajax-button.ui-state-active")
106 							.removeClass("ui-state-active");
107 				if( $(this).is('.xatajax-buttonset-multi .ui-state-active') ){ 
108 					$(this).removeClass("ui-state-active"); 
109 				} else { 
110 					$(this).addClass("ui-state-active"); 
111 				}
112 			}
113 			return true;
114 		}
115 		
116 		function buttonMouseUp(){
117 			if ( !$(this).hasClass('ui-state-disabled') ){
118 			
119 				if(! $(this).is('.xatajax-buttonset-single .xatajax-button,  .xatajax-buttonset-multi .xatajax-button') ){
120 					$(this).removeClass("ui-state-active");
121 				}
122 			
123 			}
124 			return true;
125 		
126 		}
127 		
128 		/**
129 		 * Adds a button to this button group.  It overrides the 
130 		 * standard Component add method by adding some additional
131 		 * event handlers to the button.  Although this can be used
132 		 * to add other types of elements to the buttongroup, it
133 		 * isn't advisable and will yield undefined results.
134 		 * @override
135 		 * 
136 		 * @param {Button} c The button to add.
137 		 * @param {mixed} param
138 		 * @returns {ButtonGroup} Self for chaining.
139 		 *
140 		 */
141 		function add(c, param){
142 			this.getSuper(Component).add(c, param);
143 			if ( XataJax.instanceOf(c, Button) ){
144 				decorateButton(c);
145 			}
146 			return this;
147 		}
148 		
149 		/**
150 		 * Removes a button from this group.  It overrides the Component
151 		 * remove method by also removing event handlers that had been
152 		 * added on add.
153 		 * 
154 		 * @param {Button} c The button to add to the group.
155 		 * @param {boolean} repack Whether to repack the layout (not used in ButtonGroup).
156 		 * @returns {ButtonGroup} Self for chaining.
157 		 */
158 		function remove(c, repack){
159 			this.getSuper(Component).remove(c, repack);
160 			if ( XataJax.instanceOf(c, Button) ){
161 				undecorateButton(c);
162 			}
163 		}
164 		
165 		/**
166 		 * Decorates a button element.  This adds necessary event handlers
167 		 * to make the button work nicely in this button group.
168 		 *
169 		 * @private
170 		 * @param {Button} btn The button component to decorate.
171 		 *
172 		 */
173 		function decorateButton(btn){
174 			//return;
175 			$(btn.getElement())
176 				.bind('mousedown.ButtonGroup',buttonMouseDown)
177 				.bind('mouseup.ButtonGroup', buttonMouseUp);
178 			
179 		}
180 		
181 		/**
182 		 * Undecorates a button.  This removes the event handlers that were added
183 		 * by decorateButton.
184 		 *
185 		 * @private
186 		 * @param {Button} btn Teh button to undecorate.
187 		 */
188 		function undecorateButton(btn){
189 			//return;
190 			$(btn.getElement())
191 				.unbind('mousedown.ButtonGroup', buttonMouseDown)
192 				.unbind('mouseup.ButtonGroup', buttonMouseUp);
193 		}
194 	}
195 })();
